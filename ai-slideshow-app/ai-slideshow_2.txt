<script>
(() => {
  'use strict';

  // ══════════════════════════════════════════════════════════════════════
  // IndexedDB — single kv-style store for image Blobs + manifest
  // ══════════════════════════════════════════════════════════════════════
  const DB_NAME = 'slideshow-db';
  const DB_STORE = 'kv';
  const MANIFEST_KEY = 'manifest';
  // 150 images at typical 768×512 JPEG sizes ≈ 15-30MB total; still well
  // under any device's IDB pressure threshold. Favorited entries are
  // protected from eviction (see cacheStore() below).
  //
  // This cap is also the hard ceiling for total in-flight work — the
  // queue runner refuses to schedule more than (MAX_IMAGES - cached -
  // active - pending) at any time, so the user can never overshoot.
  const MAX_IMAGES = 150;
  let _db = null, _dbP = null;

  function openDb() {
    if (_db) return Promise.resolve(_db);
    if (_dbP) return _dbP;
    _dbP = new Promise((resolve, reject) => {
      const req = indexedDB.open(DB_NAME, 1);
      req.onupgradeneeded = e => {
        const db = e.target.result;
        if (!db.objectStoreNames.contains(DB_STORE)) db.createObjectStore(DB_STORE);
      };
      req.onsuccess = e => { _db = e.target.result; resolve(_db); };
      req.onerror = e => { _dbP = null; reject(e.target.error); };
    });
    return _dbP;
  }
  async function idbGet(key) {
    try {
      const db = await openDb();
      return await new Promise((res, rej) => {
        const r = db.transaction(DB_STORE, 'readonly').objectStore(DB_STORE).get(key);
        r.onsuccess = () => res(r.result ?? null);
        r.onerror = e => rej(e.target.error);
      });
    } catch { return null; }
  }
  async function idbSet(key, val) {
    try {
      const db = await openDb();
      await new Promise((res, rej) => {
        const tx = db.transaction(DB_STORE, 'readwrite');
        tx.objectStore(DB_STORE).put(val, key);
        tx.oncomplete = res;
        tx.onerror = e => rej(e.target.error);
      });
    } catch {}
  }
  async function idbDel(key) {
    try {
      const db = await openDb();
      await new Promise((res, rej) => {
        const tx = db.transaction(DB_STORE, 'readwrite');
        tx.objectStore(DB_STORE).delete(key);
        tx.oncomplete = res;
        tx.onerror = e => rej(e.target.error);
      });
    } catch {}
  }

  // ══════════════════════════════════════════════════════════════════════
  // Cache — image Blobs stored individually; manifest tracks order +
  // metadata. Each entry: { id, prompt, createdAt, seed?, favorited? }
  //
  // `seed` and `favorited` are optional — entries persisted before this
  // feature shipped won't have them, so all readers use `?? null` /
  // `=== true` patterns rather than expecting them to be present.
  //
  // Eviction: when the cache is full, prefer non-favorited oldest entries
  // for eviction. Only evict a favorited entry as a last resort (every
  // slot is favorited). The new image is appended AFTER eviction so it
  // can never become its own eviction target.
  // ══════════════════════════════════════════════════════════════════════
  async function cacheStore(dataUrl, prompt, seed) {
    const id = 'img_' + Date.now() + '_' + Math.random().toString(36).slice(2, 6);
    const blob = await fetch(dataUrl).then(r => r.blob());
    const manifest = (await idbGet(MANIFEST_KEY)) || [];

    // Step 1 — save the new blob first. If anything fails after this
    // point the worst case is an orphan blob (small storage leak)
    // rather than a manifest entry pointing at a missing blob.
    await idbSet(id, blob);

    // Step 2 — decide evictions in memory. Walk oldest→newest preferring
    // non-favorited entries; only evict a favorite when literally every
    // slot is favorited.
    const evicted = [];
    while (manifest.length >= MAX_IMAGES) {
      let evictIdx = manifest.findIndex(m => !m.favorited);
      if (evictIdx === -1) {
        // Every entry is favorited — evict the absolute oldest as a
        // last resort. User can re-favorite the new one if they want.
        console.warn('[cache] all entries favorited; evicting oldest');
        evictIdx = 0;
      }
      const [removed] = manifest.splice(evictIdx, 1);
      if (removed) evicted.push(removed);
    }

    // Step 3 — append + persist manifest. After this write the cache
    // is fully consistent.
    manifest.push({
      id,
      prompt: prompt || '',
      createdAt: Date.now(),
      seed: (typeof seed === 'number' && Number.isFinite(seed)) ? seed : null,
      favorited: false,
    });
    await idbSet(MANIFEST_KEY, manifest);

    // Step 4 — best-effort cleanup of evicted blobs.
    for (const old of evicted) {
      try { await idbDel(old.id); }
      catch (e) { console.warn('[cache] failed to delete evicted blob', old.id, e); }
    }
    return id;
  }
  const getManifest = () => idbGet(MANIFEST_KEY).then(m => m || []);
  async function cacheGetUrl(id) {
    const blob = await idbGet(id);
    return blob ? URL.createObjectURL(blob) : null;
  }
  async function cacheDelete(id) {
    await idbDel(id);
    const m = (await getManifest()).filter(x => x.id !== id);
    await idbSet(MANIFEST_KEY, m);
  }
  async function cacheClear() {
    const m = await getManifest();
    for (const e of m) await idbDel(e.id);
    await idbSet(MANIFEST_KEY, []);
  }

  /** Toggle the `favorited` flag on a manifest entry. Returns the new
   *  value, or null if the entry wasn't found. */
  async function cacheToggleFavorite(id) {
    const m = await getManifest();
    const entry = m.find(e => e.id === id);
    if (!entry) return null;
    entry.favorited = !entry.favorited;
    await idbSet(MANIFEST_KEY, m);
    return entry.favorited;
  }

  /** Return a sorted/filtered VIEW of the manifest. Doesn't mutate
   *  storage. Modes: newest | oldest | favorites | seed.
   *  When favOnly is true, drop non-favorited entries entirely. */
  function sortManifest(manifest, mode, favOnly) {
    let view = (manifest || []).slice();
    if (favOnly) view = view.filter(m => m.favorited === true);
    switch (mode) {
      case 'oldest':
        view.sort((a, b) => (a.createdAt || 0) - (b.createdAt || 0));
        break;
      case 'favorites':
        // Favorites first; within each group, newest first.
        view.sort((a, b) => {
          const fa = a.favorited === true ? 1 : 0;
          const fb = b.favorited === true ? 1 : 0;
          if (fa !== fb) return fb - fa;
          return (b.createdAt || 0) - (a.createdAt || 0);
        });
        break;
      case 'seed':
        // Numeric seed sort; entries without a seed sink to the bottom.
        view.sort((a, b) => {
          const sa = (typeof a.seed === 'number') ? a.seed : Infinity;
          const sb = (typeof b.seed === 'number') ? b.seed : Infinity;
          if (sa !== sb) return sa - sb;
          return (b.createdAt || 0) - (a.createdAt || 0);
        });
        break;
      case 'newest':
      default:
        view.sort((a, b) => (b.createdAt || 0) - (a.createdAt || 0));
        break;
    }
    return view;
  }

  /** Per-id click debounce — prevents rapid-fire clicks from queuing
   *  duplicate downloads of the same image. Cleared 1.5s after click. */
  const _downloadInflight = new Set();
  function _extFromMime(mime) {
    // Perchance's text-to-image plugin defaults to JPEG. Older blobs
    // may be PNG. Detect from the blob's actual type rather than
    // guessing — saves the user from a wrongly-extensioned file.
    if (!mime || typeof mime !== 'string') return 'jpg';
    const m = mime.toLowerCase();
    if (m.includes('jpeg') || m.includes('jpg')) return 'jpg';
    if (m.includes('png')) return 'png';
    if (m.includes('webp')) return 'webp';
    if (m.includes('gif')) return 'gif';
    return 'jpg';
  }
  async function cacheDownload(id) {
    if (_downloadInflight.has(id)) return; // debounce
    _downloadInflight.add(id);
    setTimeout(() => _downloadInflight.delete(id), 1500);
    const blob = await idbGet(id);
    if (!blob) return;
    const ext = _extFromMime(blob.type);
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = id + '.' + ext;
    a.click();
    setTimeout(() => URL.revokeObjectURL(a.href), 5000);
  }

  // ══════════════════════════════════════════════════════════════════════
  // Settings persistence — single `settings` key in the same kv store.
  // Covers: prompt text (pos/neg), generator params, playback params,
  // enabled transitions, order mode, auto-gen toggle, active style
  // chip ids, gallery sort mode, favorites-only filter, enhance length.
  //
  // Writes are debounced so rapid slider / typing events don't thrash IDB.
  // ══════════════════════════════════════════════════════════════════════
  const SETTINGS_KEY = 'settings';

  // Active state for things that don't bind directly to a DOM element
  // and need a JS-side variable to round-trip through saveSettings().
  let activeStyleIds = [];          // string[] — currently-toggled style chip ids
  let gallerySort = 'newest';       // 'newest' | 'oldest' | 'favorites' | 'seed'
  let favOnly = false;              // favorites-only filter

  // DEFAULT_SETTINGS is materialised lazily (after TRANSITIONS is declared
  // below) because it references DEFAULT_ENABLED which lives in the
  // transitions section. We merge with these on load to forward-compat
  // new fields with old saved states.
  function buildDefaultSettings() {
    return {
      promptPos:          '',
      promptNeg:          '',
      orientation:        'landscape',
      guidance:           7,
      steps:              30,
      interval:           8,
      duration:           2,
      order:              'sequential',
      autoGen:            false,
      enabledTransitions: Array.from(DEFAULT_ENABLED),
      activeStyleIds:     [],
      gallerySort:        'newest',
      favOnly:            false,
      enhanceLength:      'medium',
    };
  }

  async function loadSettings() {
    const saved = (await idbGet(SETTINGS_KEY)) || {};
    const s = Object.assign(buildDefaultSettings(), saved);

    // Inputs
    $('promptPos').value = s.promptPos || '';
    $('promptNeg').value = s.promptNeg || '';
    $('orientation').value = s.orientation;
    if ($('enhanceLength')) $('enhanceLength').value = s.enhanceLength || 'medium';

    // Sliders + their value readouts
    $('guidance').value = s.guidance;
    $('guidanceVal').textContent = s.guidance;
    $('steps').value = s.steps;
    $('stepsVal').textContent = s.steps;
    $('interval').value = s.interval;
    $('intervalVal').textContent = s.interval;
    $('duration').value = s.duration;
    $('durationVal').textContent = Number(s.duration).toFixed(1);

    // Auto-gen checkbox
    $('autoGen').checked = !!s.autoGen;

  // Generator — parallel-capable, clamped to MAX_PARALLEL (= Perchance's limit)
  // ══════════════════════════════════════════════════════════════════════
  const MAX_PARALLEL = 4;
  const RESOLUTIONS = {
    default:   null,           // plugin picks
    square:    '512x512',

    landscape: '768x512',

  /** Compose the final generation params from the user's textareas plus
   *  any active style chips. Pure — no side effects.
   *
   *    1. user prompt              → primary intent (always first)
   *    2. active preset fragments  → additive, appended in selection order
   *    3. fallback defaults        → only when nothing else is set
   *
   *  Each preset contributes BOTH a prompt and a negative fragment, so
   *  e.g. picking "Misty" automatically negates "sharp clarity, hard
   *  edges". Empty pieces are filtered out so we never emit a stray
   *  leading/trailing comma.
   *
   *  A fresh client-side seed is generated each call (signed-32-bit
   *  positive) and stored on the manifest entry so the gallery can sort
   *  by it and a future "regen with same seed" feature has the value. */
  function currentParams() {
    const userPrompt = ($('promptPos').value || '').trim();
    const userNeg = ($('promptNeg').value || '').trim();
    const activePresets = activeStyleIds.map(id => STYLE_PRESET_MAP[id]).filter(Boolean);

    const promptParts = [userPrompt, ...activePresets.map(p => p.prompt)].filter(Boolean);
    const finalPrompt = promptParts.length
      ? promptParts.join(', ')
      : 'abstract dreamlike ethereal, soft focus';

    const negParts = [userNeg, ...activePresets.map(p => p.negative).filter(Boolean)].filter(Boolean);
    const finalNeg = negParts.length
      ? negParts.join(', ')
      : 'text, watermark, deformed';

    const seed = Math.floor(Math.random() * 2147483647);

    return {
      prompt: finalPrompt,
      negative: finalNeg,
      resolution: RESOLUTIONS[$('orientation').value] || null,
      guidance: Number($('guidance').value),
      steps: Number($('steps').value),
      seed,
    };
  }

  async function generateOne() {
    if (typeof root === 'undefined' || !root.textToImagePlugin) {
      toast('textToImagePlugin not loaded. Check the top-editor DSL.', 'err');
      return null;
    }
    // Cooperative queueing so 5+ callers don't all hammer the plugin at once
    while (activeGens >= MAX_PARALLEL) await sleep(150);
    activeGens++;
    try {
      const p = currentParams();
      // The text-to-image-plugin's documented contract is {prompt,
      // negative, style}. Sending undocumented extras like
      // numInferenceSteps / guidanceScale / seed / resolution routes
      // their values through Perchance's DSL evaluator, which parses
      // `{listName}` as a list reference and throws on
      // brace-containing user input. We pass ONLY the documented
      // fields here, and capture the seed locally for the manifest.
      // (HNE makes the same call shape; see bgEngine.generateOne.)
      const call = {
        prompt: p.prompt,
        negative: p.negative,
      };
      const result = await root.textToImagePlugin(call);
      const dataUrl = result?.dataUrl || String(result);
      if (!dataUrl || !dataUrl.startsWith('data:')) return null;
      const id = await cacheStore(dataUrl, p.prompt, p.seed);
      _bumpQueueUI();
      return id;
    } catch (e) {
      console.error('generateOne:', e);
      return null;
    } finally {
      activeGens--;
      _bumpQueueUI();
    }
  }

  async function generateBatch(n) {
    const cap = Math.min(Math.max(1, n), MAX_PARALLEL);
    return Promise.all(Array.from({ length: cap }, () => generateOne()));
  }

  async function renderGallery() {
    const grid = $('gallery');
    const count = $('galleryCount');
    const info = $('galleryInfo');
    if (!grid) return;

    const myGen = ++_galleryRenderGen;
    const manifest = await getManifest();
    if (myGen !== _galleryRenderGen) return; // newer render started

    if (count) count.textContent = `(${manifest.length})`;
    _lastKnownCacheCount = manifest.length;
    // Refresh the +1/+5 enabled state in case a delete just freed headroom.
    _bumpQueueUI();

    // Sync toolbar controls — opening the panel from anywhere reflects
    // the persisted state.
    if ($('gallerySort') && $('gallerySort').value !== gallerySort) {
      $('gallerySort').value = gallerySort;
    }
    if ($('favFilter')) {
      $('favFilter').classList.toggle('active', favOnly);
      $('favFilter').setAttribute('aria-pressed', favOnly ? 'true' : 'false');
    }

    const view = sortManifest(manifest, gallerySort, favOnly);
    const favCount = manifest.filter(m => m.favorited === true).length;

    if (info) {
      const totalStr = `${manifest.length} image${manifest.length === 1 ? '' : 's'}`;
      const favStr = favCount > 0 ? ` · ${favCount} favorite${favCount === 1 ? '' : 's'}` : '';
      const filterStr = favOnly && favCount !== manifest.length ? ` · showing ${view.length}` : '';
      info.textContent = totalStr + favStr + filterStr;
    }

    if (!view.length) {
      grid.innerHTML = manifest.length === 0
        ? '<div class="gallery-empty">No images yet. Generate some above.</div>'
        : '<div class="gallery-empty">No images match the current filter.</div>';
      return;
    }

    // and data-idx (relative to the current sorted view) so clicking
    // opens the lightbox at the right position.
    grid.innerHTML = view.map((c, idx) => {
      const seedTag = (typeof c.seed === 'number') ? `<div class="seed-tag" title="seed">#${c.seed}</div>` : '';
      const favClass = c.favorited === true ? ' is-fav' : '';
      const favActive = c.favorited === true ? ' active' : '';
      const favIcon = c.favorited === true ? '★' : '☆';
      const tip = c.prompt ? c.prompt.replace(/"/g, '&quot;').slice(0, 200) : 'click to view';
      return `<div class="gallery-item${favClass}" data-id="${c.id}" data-idx="${idx}" title="${tip}">

        ${seedTag}
      </div>`;
    }).join('');

    // Lazy-load thumbnails — bail per-iteration if a newer render started
    // OR if the matching DOM node is gone (item was deleted concurrently).
    for (const entry of view) {
      if (myGen !== _galleryRenderGen) return;
      const url = await cacheGetUrl(entry.id);
      if (myGen !== _galleryRenderGen) {
        if (url) URL.revokeObjectURL(url); // not going to use this — release
        return;
      }
      const img = grid.querySelector(`.gallery-item[data-id="${entry.id}"] img`);
      if (img && url) {
        img.src = url;
      } else if (url) {
        URL.revokeObjectURL(url); // item vanished mid-fetch
      }
    }

    // Per-button handlers — stop propagation so a button click doesn't
    // also trigger the item-body's lightbox-open click.
    grid.querySelectorAll('.del').forEach(btn => {

        e.stopPropagation();

      });

    grid.querySelectorAll('.dl').forEach(btn => {

        e.stopPropagation();
        cacheDownload(btn.dataset.id);
      });
    });

      btn.addEventListener('click', async (e) => {
        e.stopPropagation();
        await cacheToggleFavorite(btn.dataset.id);
        renderGallery();
      });
    });
    // Click on the item body (not on a button) opens the lightbox at this
    // position in the SORTED view — so the lightbox's prev/next walk in
    // the same order the user is looking at.
    grid.querySelectorAll('.gallery-item').forEach(itemEl => {
      itemEl.addEventListener('click', (e) => {
        if (e.target.closest('button')) return;
        const idx = Number(itemEl.dataset.idx);
        if (Number.isFinite(idx)) lightbox.open(view, idx);
      });
    });
  }

  // Style chip UI — additive prompt modifiers (no textarea mutation)
  // ══════════════════════════════════════════════════════════════════════
  function buildStyleChips() {
    const root = $('styleChips');
    const countEl = $('chipCount');
    const clearBtn = $('chipsClear');
    if (!root) return;

    const active = new Set(activeStyleIds);
    // Group by category for one row per category
    const byCat = {};
    for (const cat of STYLE_CATEGORIES) byCat[cat.id] = [];
    for (const p of STYLE_PRESETS) {
      if (!byCat[p.cat]) byCat[p.cat] = [];
      byCat[p.cat].push(p);
    }
    const rows = STYLE_CATEGORIES.map(cat => {
      const items = byCat[cat.id] || [];
      if (!items.length) return '';
      const chips = items.map(p => {
        const cls = active.has(p.id) ? 'style-chip active' : 'style-chip';
        const tip = (p.prompt || '').replace(/"/g, '&quot;');
        return `<button type="button" class="${cls}" data-style-id="${p.id}" title="${tip}" aria-pressed="${active.has(p.id) ? 'true' : 'false'}">${p.label}</button>`;
      }).join('');
      return `<div class="chip-cat-row"><span class="chip-cat-label">${cat.label}</span>${chips}</div>`;
    }).join('');
    root.innerHTML = rows;

    if (countEl) countEl.textContent = active.size > 0 ? `· ${active.size} active` : '';
    if (clearBtn) clearBtn.disabled = active.size === 0;

    // Wire each chip — rebuild the whole row on each click rather than
    // mutating in place (simpler, and the row is tiny).
    root.querySelectorAll('.style-chip').forEach(chip => {
      chip.addEventListener('click', () => {
        const id = chip.dataset.styleId;
        if (!id) return;
        const cur = new Set(activeStyleIds);
        if (cur.has(id)) cur.delete(id); else cur.add(id);
        activeStyleIds = Array.from(cur);
        buildStyleChips();
        saveSettings(0);
      });
    });
  }

  // ══════════════════════════════════════════════════════════════════════
  // Prompt enhancer — uses aiTextPlugin to rewrite the user's casual
  // input into an optimized AI image generation prompt at the requested
  // verbosity. Streams back into the textarea. Click again to cancel.
  //
  // Adapted from HNE's BG_ENHANCE_INSTRUCTIONS (which were in turn
  // adapted from promptyx). The instructions are intentionally strict
  // about format ("ONLY the final raw prompt string, no preamble") so
  // the streamed result drops straight into the textarea unedited.
  // ══════════════════════════════════════════════════════════════════════
  const ENHANCE_INSTRUCTIONS = {
    short: `ROLE: Expert AI image prompt engineer.
ACTION: Transform the input into a concise, optimized AI image generation prompt (300-600 chars).
FORMAT: Comma-separated precise visual descriptors only.
ENHANCE WITH: specific lighting type, camera angle/lens, primary color palette, key material/texture detail, mood/atmosphere word, one compositional technique.
FORBIDDEN: 8K, UHD, hyperrealism, "extreme detail", explanatory sentences, question marks, quotes around the result.
OUTPUT: ONLY the final raw prompt string. No preamble. No quotes. No "Here's your prompt:" or similar.`,
    medium: `ROLE: Expert AI image prompt engineer.
ACTION: Transform the input into a richly-detailed AI image generation prompt (600-1100 chars).
FORMAT: Comma-separated visual descriptors with moderate depth.
ENHANCE WITH: specific lighting setup, focal length/camera type, 2-3 color palette descriptors, multiple surface textures, spatial composition (foreground/midground/background), atmosphere/weather, time of day, style anchor (one artist or aesthetic reference).
FORBIDDEN: 8K, UHD, hyperrealism, "extreme detail", narration, instructional language, quotes around the result.
OUTPUT: ONLY the final raw prompt string. No preamble. No quotes.`,
    long: `ROLE: Expert AI image prompt engineer.
ACTION: Transform the input into a comprehensive, cinematic-quality AI image generation prompt (1100-1600 chars).
FORMAT: Flowing comma-separated visual descriptors at maximum fidelity.
ENHANCE WITH: complex multi-source lighting rig, specific lens + aperture + film stock, exhaustive color grading descriptor, layered texture details (micro and macro), full environmental world-building (weather + time + season + location), subject details, composition framing technique, emotional/psychological undertone, 1-2 specific artist/film/art-movement references.
FORBIDDEN: 8K, UHD, hyperrealism, "extreme detail", storytelling prose, instructional text, quotes around the result.
OUTPUT: ONLY the final raw prompt string. No preamble. No quotes.`,
  };

  let _enhanceToken = 0;
  let _enhanceRunning = false;

  async function runEnhance() {
    const btn = $('enhanceBtn');
    const input = $('promptPos');
    if (!btn || !input) return;

    // Toggle-cancel if already running — increment token so the in-flight
    // call sees a stale token and bails on its next chunk.
    if (_enhanceRunning) {
      _enhanceToken++;
      _enhanceRunning = false;
      btn.disabled = false;
      btn.textContent = '✨ enhance';
      return;
    }

    if (typeof root === 'undefined' || !root.aiTextPlugin) {
      toast('Prompt enhancer needs aiTextPlugin in the top-editor DSL.', 'warn');
      return;
    }

    const userInput = (input.value || '').trim();
    if (!userInput) {
      toast('Type a casual idea first — I\'ll engineer it into an optimized image prompt.', 'warn');
      return;
    }

    const length = $('enhanceLength')?.value || 'medium';
    const instr = ENHANCE_INSTRUCTIONS[length] || ENHANCE_INSTRUCTIONS.medium;
    const myToken = ++_enhanceToken;
    _enhanceRunning = true;
    btn.textContent = '⏹ stop';

    const original = userInput;
    input.value = '';
    let acc = '';

    try {
      const streamObj = root.aiTextPlugin({
        instruction: `${instr}\n\nUser Input: ${original}`,
        onChunk: (data) => {
          if (myToken !== _enhanceToken) {
            try { streamObj?.stop?.(); } catch {}
            throw new Error('cancelled');
          }
          if (data.isFromStartWith) return;
          acc += data.textChunk || '';
          input.value = acc;
        },
      });
      const result = await streamObj;
      if (myToken !== _enhanceToken) return;

      // Some flows return the whole text on result.generatedText
      // without ever firing onChunk; pick whichever's non-empty.
      const finalText = (acc || result?.generatedText || '').trim();
      if (!finalText) {
        input.value = original;
        toast('Enhance returned nothing — your original prompt was kept.', 'warn');
        return;
      }
      // Strip stray surrounding quotes some models add despite instructions
      const cleaned = finalText.replace(/^["']|["']$/g, '');
      input.value = cleaned;
      toast(`Prompt enhanced (${length}).`);
    } catch (e) {
      if (e?.message === 'cancelled') {
        input.value = original;
        toast('Enhance cancelled — original prompt restored.');
      } else {
        input.value = original;
        console.warn('[enhance] failed:', e);
        toast('Enhance failed — your original prompt was kept.', 'warn');
      }
    } finally {
      if (myToken === _enhanceToken) {
        _enhanceRunning = false;
        btn.disabled = false;
        btn.textContent = '✨ enhance';
        saveSettings(0);
      }
    }
  }

  // ══════════════════════════════════════════════════════════════════════
  // Lightbox preview popup + Fullscreen zoom/pan viewer
  // ══════════════════════════════════════════════════════════════════════
  // Two-tier viewer:
  //   • Lightbox (z:110) — compact preview with metadata + nav buttons
  //   • Fullscreen (z:120) — wheel/pinch zoom + drag pan + swipe nav
  // Closing fullscreen returns to the lightbox at the same view/index;
  // closing the lightbox closes both.
  //
  // State lives on this object (not in saved settings) — every reopen
  // starts fresh.
  const lightbox = {
    view: [],
    index: 0,
    _bound: false,
    _fsBound: false,
    _fsScale: 1,
    _fsPanX: 0, _fsPanY: 0,
    _fsMinScale: 1,
    _fsMaxScale: 8,
    _fsDrag: null,
    _fsPinch: null,
    _fsSwipe: null,

    open(view, index) {
      if (!Array.isArray(view) || view.length === 0) return;
      this.view = view;
      this.index = Math.max(0, Math.min(index | 0, view.length - 1));
      this._bindOnce();
      const modal = $('lightboxModal');
      if (modal) modal.classList.add('visible');
      document.body.style.overflow = 'hidden';
      this._loadCurrent();
    },
    close() {
      this.exitFullscreen();
      const modal = $('lightboxModal');
      if (modal) modal.classList.remove('visible');
      document.body.style.overflow = '';
      for (const id of ['lightboxImg', 'fsImg']) {
        const img = $(id);
        if (img && img.src && img.src.startsWith('blob:')) {
          try { URL.revokeObjectURL(img.src); } catch {}
          img.src = '';
        }
      }
    },
    next() {
      if (this.index >= this.view.length - 1) return;
      this.index++;
      this._loadCurrent();
      if (this._isFullscreen()) this._fsLoadCurrent();
    },
    prev() {
      if (this.index <= 0) return;
      this.index--;
      this._loadCurrent();
      if (this._isFullscreen()) this._fsLoadCurrent();
    },

    async _loadCurrent() {
      const entry = this.view[this.index];
      if (!entry) return;
      const img = $('lightboxImg');
      const meta = $('lightboxMeta');
      const promptEl = $('lightboxPrompt');
      const favBtn = $('lightboxFavorite');
      const prevBtn = $('lightboxPrev');
      const nextBtn = $('lightboxNext');

      if (img && img.src && img.src.startsWith('blob:')) {
        try { URL.revokeObjectURL(img.src); } catch {}
      }
      const url = await cacheGetUrl(entry.id);
      if (img && url) img.src = url;

      if (meta) {
        const parts = [`${this.index + 1} / ${this.view.length}`];
        if (typeof entry.seed === 'number') parts.push(`seed ${entry.seed}`);
        if (entry.createdAt) {
          const d = new Date(entry.createdAt);
          parts.push(d.toLocaleDateString() + ' ' + d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }));
        }
        meta.textContent = parts.join(' · ');
      }
      if (promptEl) {
        const p = entry.prompt || '';
        promptEl.textContent = p.length > 240 ? p.slice(0, 237) + '…' : p;
      }
      if (favBtn) {
        favBtn.classList.toggle('active', entry.favorited === true);
        favBtn.textContent = entry.favorited === true ? '★ favorited' : '☆ favorite';
        favBtn.title = entry.favorited === true ? 'unfavorite (f)' : 'favorite (f)';
      }
      if (prevBtn) prevBtn.disabled = this.index <= 0;
      if (nextBtn) nextBtn.disabled = this.index >= this.view.length - 1;
    },

    enterFullscreen() {
      const fs = $('fullscreenModal');
      if (!fs) return;
      this._bindFullscreenOnce();
      fs.classList.add('visible');
      this._fsResetZoom();
      this._fsLoadCurrent();
    },
    exitFullscreen() {
      const fs = $('fullscreenModal');
      if (!fs || !fs.classList.contains('visible')) return;
      fs.classList.remove('visible');
      const fsImg = $('fsImg');
      if (fsImg && fsImg.src && fsImg.src.startsWith('blob:')) {
        try { URL.revokeObjectURL(fsImg.src); } catch {}
        fsImg.src = '';
      }
      this._fsDrag = null;
      this._fsPinch = null;
      this._fsSwipe = null;
    },
    _isFullscreen() {
      return $('fullscreenModal')?.classList.contains('visible') === true;
    },

    async _fsLoadCurrent() {
      const entry = this.view[this.index];
      if (!entry) return;
      const img = $('fsImg');
      const meta = $('fsMeta');
      const prevBtn = $('fsPrev');
      const nextBtn = $('fsNext');
      if (img && img.src && img.src.startsWith('blob:')) {
        try { URL.revokeObjectURL(img.src); } catch {}
      }
      const url = await cacheGetUrl(entry.id);
      if (img && url) {
        img.src = url;
        // Suppress the transition during image swap so reset-zoom doesn't
        // animate the new image's transform from the old image's matrix.
        img.classList.add('no-transition');
        requestAnimationFrame(() => {
          this._fsResetZoom();
          requestAnimationFrame(() => img.classList.remove('no-transition'));
        });
      }
      if (meta) {
        const parts = [`${this.index + 1} / ${this.view.length}`];
        if (typeof entry.seed === 'number') parts.push(`seed ${entry.seed}`);
        meta.textContent = parts.join(' · ');
      }
      if (prevBtn) prevBtn.disabled = this.index <= 0;
      if (nextBtn) nextBtn.disabled = this.index >= this.view.length - 1;
    },

    _fsApplyTransform() {
      const img = $('fsImg');
      if (!img) return;
      img.style.transform = `translate(${this._fsPanX}px, ${this._fsPanY}px) scale(${this._fsScale})`;
      const lbl = $('fsZoomLabel');
      if (lbl) lbl.textContent = `${Math.round(this._fsScale * 100)}%`;
    },
    _fsResetZoom() {
      this._fsScale = 1; this._fsPanX = 0; this._fsPanY = 0;
      this._fsApplyTransform();
    },
    /** Zoom around a screen-space focal point — keeps the pixel under
     *  the cursor anchored. */
    _fsZoomAt(factor, focalX, focalY) {
      const newScale = Math.max(this._fsMinScale, Math.min(this._fsMaxScale, this._fsScale * factor));
      if (newScale === this._fsScale) return;
      const stage = $('fsStage');
      if (!stage) { this._fsScale = newScale; this._fsApplyTransform(); return; }
      const rect = stage.getBoundingClientRect();
      const sx = focalX - rect.left - rect.width / 2;
      const sy = focalY - rect.top  - rect.height / 2;
      const ratio = newScale / this._fsScale;
      this._fsPanX = sx - (sx - this._fsPanX) * ratio;
      this._fsPanY = sy - (sy - this._fsPanY) * ratio;
      this._fsScale = newScale;
      this._fsApplyTransform();
    },

    _bindOnce() {
      if (this._bound) return;
      this._bound = true;
      const self = this;
      const modal = $('lightboxModal');

      $('lightboxClose')?.addEventListener('click', () => self.close());
      $('lightboxPrev') ?.addEventListener('click', () => self.prev());
      $('lightboxNext') ?.addEventListener('click', () => self.next());
      $('lightboxFullscreen')?.addEventListener('click', () => self.enterFullscreen());
      $('lightboxFavorite')?.addEventListener('click', async () => {
        const entry = self.view[self.index];
        if (!entry) return;
        const newVal = await cacheToggleFavorite(entry.id);
        if (newVal !== null) {
          entry.favorited = newVal;
          self._loadCurrent();
          renderGallery();
        }
      });
      $('lightboxDelete')?.addEventListener('click', async () => {
        const entry = self.view[self.index];
        if (!entry) return;
        await cacheDelete(entry.id);
        self.view.splice(self.index, 1);
        if (self.view.length === 0) { renderGallery(); self.close(); return; }
        if (self.index >= self.view.length) self.index = self.view.length - 1;
        self._loadCurrent();
        if (self._isFullscreen()) self._fsLoadCurrent();
        renderGallery();
      });

      // Click backdrop (not the card) closes — but only when fullscreen
      // isn't on top of us.
      modal?.addEventListener('click', (e) => {
        if (self._isFullscreen()) return;
        if (e.target === modal) self.close();
      });

      // Keyboard nav — fullscreen owns its own keys when visible
      document.addEventListener('keydown', (e) => {
        if (self._isFullscreen()) return;
        if (!modal?.classList.contains('visible')) return;
        switch (e.key) {
          case 'Escape':     e.preventDefault(); self.close(); break;
          case 'ArrowLeft':  e.preventDefault(); self.prev();  break;
          case 'ArrowRight': e.preventDefault(); self.next();  break;
          case 'Enter':      e.preventDefault(); self.enterFullscreen(); break;
          case 'f': case 'F': e.preventDefault();
            $('lightboxFavorite')?.click(); break;
          case 'Delete': case 'Backspace': e.preventDefault();
            $('lightboxDelete')?.click(); break;
        }
      });
    },

    _bindFullscreenOnce() {
      if (this._fsBound) return;
      this._fsBound = true;
      const self = this;
      const fs = $('fullscreenModal');
      const stage = $('fsStage');
      const img = $('fsImg');

      $('fsClose')?.addEventListener('click', () => self.exitFullscreen());
      $('fsPrev') ?.addEventListener('click', () => self.prev());
      $('fsNext') ?.addEventListener('click', () => self.next());
      $('fsZoomIn') ?.addEventListener('click', () => self._fsZoomAt(1.25, window.innerWidth / 2, window.innerHeight / 2));
      $('fsZoomOut')?.addEventListener('click', () => self._fsZoomAt(0.8,  window.innerWidth / 2, window.innerHeight / 2));
      $('fsZoomFit')?.addEventListener('click', () => self._fsResetZoom());

      // Click backdrop exits fullscreen (returns to lightbox preview)
      fs?.addEventListener('click', (e) => {
        if (e.target === fs) self.exitFullscreen();
      });

      // Keyboard — only when fullscreen is the topmost layer
      document.addEventListener('keydown', (e) => {
        if (!self._isFullscreen()) return;
        switch (e.key) {
          case 'Escape':     e.preventDefault(); self.exitFullscreen(); break;
          case 'ArrowLeft':  e.preventDefault(); self.prev();  break;
          case 'ArrowRight': e.preventDefault(); self.next();  break;
          case '+': case '=': e.preventDefault();
            self._fsZoomAt(1.25, window.innerWidth / 2, window.innerHeight / 2); break;
          case '-': case '_': e.preventDefault();
            self._fsZoomAt(0.8,  window.innerWidth / 2, window.innerHeight / 2); break;
          case '0':           e.preventDefault(); self._fsResetZoom(); break;
        }
      });

      // Mouse wheel zoom around cursor
      stage?.addEventListener('wheel', (e) => {
        e.preventDefault();
        const factor = e.deltaY < 0 ? 1.12 : 0.89;
        self._fsZoomAt(factor, e.clientX, e.clientY);
      }, { passive: false });

      // Drag to pan when zoomed in
      img?.addEventListener('mousedown', (e) => {
        if (self._fsScale <= 1.0001) return;
        e.preventDefault();
        self._fsDrag = { startX: e.clientX, startY: e.clientY, panX: self._fsPanX, panY: self._fsPanY };
        img.classList.add('dragging');
      });
      document.addEventListener('mousemove', (e) => {
        if (!self._fsDrag) return;
        self._fsPanX = self._fsDrag.panX + (e.clientX - self._fsDrag.startX);
        self._fsPanY = self._fsDrag.panY + (e.clientY - self._fsDrag.startY);
        self._fsApplyTransform();
      });
      document.addEventListener('mouseup', () => {
        if (self._fsDrag) { self._fsDrag = null; img?.classList.remove('dragging'); }
      });

      // Touch — single touch swipes (when at fit) OR pans (when zoomed),
      // two touches pinch-zoom around midpoint.
      const SWIPE_THRESHOLD = 50;
      const SWIPE_AXIS_RATIO = 1.4;
      stage?.addEventListener('touchstart', (e) => {
        if (e.touches.length === 1) {
          const t = e.touches[0];
          if (self._fsScale > 1.0001) {
            self._fsDrag = { startX: t.clientX, startY: t.clientY, panX: self._fsPanX, panY: self._fsPanY };
            self._fsSwipe = null;
          } else {
            self._fsSwipe = { startX: t.clientX, startY: t.clientY };
            self._fsDrag = null;
          }
        } else if (e.touches.length === 2) {
          self._fsDrag = null; self._fsSwipe = null;
          const [a, b] = e.touches;
          const dx = b.clientX - a.clientX, dy = b.clientY - a.clientY;
          self._fsPinch = { startDist: Math.hypot(dx, dy), startScale: self._fsScale };
        }
      }, { passive: true });
      stage?.addEventListener('touchmove', (e) => {
        if (e.touches.length === 1 && self._fsDrag) {
          const t = e.touches[0];
          self._fsPanX = self._fsDrag.panX + (t.clientX - self._fsDrag.startX);
          self._fsPanY = self._fsDrag.panY + (t.clientY - self._fsDrag.startY);
          self._fsApplyTransform();
          e.preventDefault();
        } else if (e.touches.length === 2 && self._fsPinch) {
          const [a, b] = e.touches;
          const dist = Math.hypot(b.clientX - a.clientX, b.clientY - a.clientY);
          const target = self._fsPinch.startScale * (dist / self._fsPinch.startDist);
          const midX = (a.clientX + b.clientX) / 2;
          const midY = (a.clientY + b.clientY) / 2;
          self._fsZoomAt(target / self._fsScale, midX, midY);
          e.preventDefault();
        }
      }, { passive: false });
      stage?.addEventListener('touchend', (e) => {
        if (self._fsSwipe && e.changedTouches.length > 0) {
          const t = e.changedTouches[0];
          const dx = t.clientX - self._fsSwipe.startX;
          const dy = t.clientY - self._fsSwipe.startY;
          if (Math.abs(dx) >= SWIPE_THRESHOLD && Math.abs(dx) >= Math.abs(dy) * SWIPE_AXIS_RATIO) {
            if (dx < 0) self.next(); else self.prev();
          }
        }
        if (e.touches.length === 0) { self._fsDrag = null; self._fsPinch = null; self._fsSwipe = null; }
      });

      // Double-click toggles between fit and 2× — quick zoom
      img?.addEventListener('dblclick', (e) => {
        if (self._fsScale > 1.0001) self._fsResetZoom();
        else self._fsZoomAt(2, e.clientX, e.clientY);
      });
    },
  };

  // ══════════════════════════════════════════════════════════════════════
  // Generation queue — split pending vs active, hard-capped at MAX_IMAGES
  // ══════════════════════════════════════════════════════════════════════
  // The +1 / +5 buttons add to a pending counter; a runner drains the
  // pending queue by launching up to MAX_PARALLEL generations at a time.
  // The status bar shows both counters separately so the user can tell
  // "queued but not started" from "currently generating".
  //
  // Hard cap: gallery + pending + active is never allowed to exceed
  // MAX_IMAGES. Every queue-add intent is clamped against the
  // currently-cached size. _qAdd() returns the number actually accepted
  // so the UI can show "added 3 of 5 (cap reached)" rather than silently
  // doing the wrong thing. Without this clamp the queue can outrun the
  // cache cap — when those generations finish, cacheStore() would
  // start evicting older (potentially favorited) entries to make room.
  let _qPending = 0;
  let _qRunning = false;

  function _bumpQueueUI() {
    const status = $('queueStatus');
    const qRemove = $('qRemove');
    const qClear = $('qClear');
    const total = _qPending + activeGens;
    if (total > 0) {
      const parts = [];
      if (_qPending > 0) parts.push(`${_qPending} queued`);
      if (activeGens > 0) parts.push(`${activeGens} generating`);
      if (status) {
        status.textContent = parts.join(' · ');
        status.classList.remove('hidden');
      }
      if (qRemove) qRemove.style.display = _qPending > 0 ? '' : 'none';
      if (qClear)  qClear.style.display  = _qPending > 0 ? '' : 'none';
    } else {
      if (status) status.classList.add('hidden');
      if (qRemove) qRemove.style.display = 'none';
      if (qClear)  qClear.style.display  = 'none';
    }
    // Disable +1 / +5 when total work would exceed the cap. We can't
    // know the cached count synchronously without another IDB hit, but
    // we use the last-known count + active + pending as a proxy. The
    // accurate clamp happens inside _qAdd(); we leave +5 enabled with
    // 1+ slots available so it can partially-accept and warn the user
    // rather than hiding the option entirely.
    const gen1 = $('gen1'); const gen5 = $('gen5');
    const knownTotal = _lastKnownCacheCount + _qPending + activeGens;
    const atCap = knownTotal >= MAX_IMAGES;
    if (gen1) gen1.disabled = atCap;
    if (gen5) gen5.disabled = atCap;
  }

  // Last-known cached image count, refreshed on every render and on
  // every queue add. Lets _bumpQueueUI() reason about cap-headroom
  // without an IDB read on every hover.
  let _lastKnownCacheCount = 0;

  async function _qRun() {
    if (_qRunning) return;
    _qRunning = true;
    while (_qPending > 0) {
      const batch = Math.min(_qPending, MAX_PARALLEL - activeGens);
      if (batch <= 0) {
        await sleep(200);
        _bumpQueueUI();
        continue;
      }
      _qPending -= batch;
      _bumpQueueUI();
      $('genStatus').textContent = `Generating ${batch} image${batch === 1 ? '' : 's'}…`;
      const promises = Array.from({ length: batch }, () =>
        generateOne().then(id => {
          _bumpQueueUI();
          if (id) renderGallery(); // progressive — each image lands as it arrives
          return id;
        })
      );
      await Promise.all(promises);
      const m = await getManifest();
      _lastKnownCacheCount = m.length;
      const remaining = _qPending + activeGens;
      $('genStatus').textContent = remaining > 0
        ? `${m.length} image${m.length === 1 ? '' : 's'} in cache. ${remaining} remaining…`
        : `Done. ${m.length} image${m.length === 1 ? '' : 's'} in cache.`;
      renderGallery();
    }
    _bumpQueueUI();
    _qRunning = false;
  }

  /** Try to queue `n` more generations. Clamps against the hard cap of
   *  MAX_IMAGES total (cached + pending + active). Returns the number
   *  actually accepted, so the caller can warn the user when their ask
   *  was partially or fully refused. */
  async function _qAdd(n) {
    const m = await getManifest();
    _lastKnownCacheCount = m.length;
    const inFlight = _qPending + activeGens;
    const headroom = Math.max(0, MAX_IMAGES - m.length - inFlight);
    const accepted = Math.min(n, headroom);
    if (accepted <= 0) {
      toast(`Cap reached — ${m.length} cached + ${inFlight} in flight = ${m.length + inFlight}/${MAX_IMAGES}. Delete some images first or favorite less.`, 'warn');
      _bumpQueueUI();
      return 0;
    }
    if (accepted < n) {
      toast(`Only queued ${accepted} of ${n} — at the ${MAX_IMAGES}-image cap.`, 'warn');
    }
    _qPending += accepted;
    _bumpQueueUI();
    _qRun();
    return accepted;
  }
  function _qRemoveOne() {
    if (_qPending > 0) {
      _qPending--;
      _bumpQueueUI();
      if (_qPending === 0 && activeGens === 0) {
        $('genStatus').textContent = 'Queue cleared.';
      }
    }
  }
  function _qClearAll() {
    if (_qPending <= 0) return;
    const dropped = _qPending;
    _qPending = 0;
    _bumpQueueUI();
    $('genStatus').textContent = `Queue cleared (${dropped} dropped). Active generations will finish.`;
    toast(`Queue cleared — ${dropped} pending dropped.`);
  }

  // ══════════════════════════════════════════════════════════════════════
  // Slideshow UI state — order (sequential|random) + enabled transitions.
  // Persisted in IndexedDB via the settings layer below.
  // ══════════════════════════════════════════════════════════════════════
  let currentOrder = 'sequential';

  // Default-enabled transitions — picked to give a pleasant mix without being
  // visually overwhelming. User can toggle individually or use the All/None buttons.
  // Keep this list small and calm — there are 30+ available, so a default of
  // 8 or so gives variety without overwhelming first-time users.
  const DEFAULT_ENABLED = new Set([
    'crossfade', 'fadeBlack', 'kenburns', 'breathe',
    'slideLeft', 'pushLeft', 'zoomIn', 'blur',
  ]);

  function getEnabledTransitions() {
    const boxes = document.querySelectorAll('#transitionGrid input[type="checkbox"]');
    return Array.from(boxes).filter(b => b.checked).map(b => b.value);
  }

  function buildTransitionGrid() {
    const grid = $('transitionGrid');
    if (!grid) return;
    const keys = Object.keys(TRANSITIONS);
    grid.innerHTML = keys.map(k => {
      const checked = DEFAULT_ENABLED.has(k) ? ' checked' : '';
      return `<label>
        <input type="checkbox" value="${k}"${checked}>
        <span>${TRANSITIONS[k].label}</span>
      </label>`;
    }).join('');
  }

  // ══════════════════════════════════════════════════════════════════════
  // Transition registry — each transition is a function that performs the
  // visual swap from oldEl → newEl. Called with (oldEl, newEl, url, durSec).
  //
  // Standard contract:
  //   • Sets newEl's backgroundImage to url
  //   • Arranges z-index so newEl paints on top during the transition
  //   • Resolves only when safe to schedule the next transition (we don't
  //     await, but the player gives each transition the interval-minus-buffer
  //     time to finish before advancing again)
  //
  // Reset pattern at start: clear prior transform/filter/clipPath/transition
  // from BOTH slides so the next transition starts from a known state.
  // ══════════════════════════════════════════════════════════════════════
  function prepSlides(oldEl, newEl) {
    // newEl is going to be the incoming slide — wipe its inline styles.
    // Includes properties the newer transitions touch (mask-image,
    // transformOrigin, backfaceVisibility) so transition leakage is
    // impossible — every run starts from a known-clean state.
    newEl.style.transition = 'none';
    newEl.style.transform = '';
    newEl.style.filter = '';
    newEl.style.clipPath = '';
    newEl.style.maskImage = '';
    newEl.style.webkitMaskImage = '';
    newEl.style.transformOrigin = '';
    newEl.style.backfaceVisibility = '';
    newEl.style.opacity = '0';
    newEl.style.zIndex = '2';
    // oldEl keeps its bg image but we ensure its z-index is below
    oldEl.style.zIndex = '1';
    // Force reflow so any subsequent transition actually animates
    void newEl.offsetWidth;
  }
  function scheduleCleanup(el, delayMs) {
    setTimeout(() => {
      // Only clear if the element isn't currently the "live" slide (another
      // transition may have promoted it in the meantime)
      if (el.style.zIndex !== '2') {
        el.style.transition = 'none';
        el.style.transform = '';
        el.style.filter = '';
        el.style.clipPath = '';
        el.style.maskImage = '';
        el.style.webkitMaskImage = '';
        el.style.transformOrigin = '';
        el.style.backfaceVisibility = '';
        el.style.opacity = '0';
        el.style.backgroundImage = '';
        el.classList.remove('show');
      }
    }, delayMs);
  }

  const TRANSITIONS = {
    cut: {
      label: 'Cut',
      run(oldEl, newEl, url, dur) {
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '1';
        newEl.classList.add('show');
        oldEl.classList.remove('show');
        oldEl.style.opacity = '0';
      },
    },
    crossfade: {
      label: 'Crossfade',
      run(oldEl, newEl, url, dur) {
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        void newEl.offsetWidth;
        newEl.style.transition = `opacity ${dur}s ease`;
        oldEl.style.transition = `opacity ${dur}s ease`;
        newEl.style.opacity = '1';
        newEl.classList.add('show');
        oldEl.style.opacity = '0';
        oldEl.classList.remove('show');
      },
    },
    fadeBlack: {
      label: 'Fade → black',
      async run(oldEl, newEl, url, dur) {
        const b = $('blackout');
        b.style.background = '#000';
        b.style.transition = `opacity ${dur / 2}s ease`;
        b.classList.add('show');
        await sleep(dur * 500);
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '1';
        newEl.classList.add('show');
        oldEl.classList.remove('show');
        oldEl.style.opacity = '0';
        await sleep(50);
        b.classList.remove('show');
      },
    },
    fadeWhite: {
      label: 'Fade → white',
      async run(oldEl, newEl, url, dur) {
        const b = $('blackout');
        b.style.background = '#fff';
        b.style.transition = `opacity ${dur / 2}s ease`;
        b.classList.add('show');
        await sleep(dur * 500);
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '1';
        newEl.classList.add('show');
        oldEl.classList.remove('show');
        oldEl.style.opacity = '0';
        await sleep(50);
        b.classList.remove('show');
        setTimeout(() => { b.style.background = '#000'; }, dur * 1000 + 300);
      },
    },
    slideLeft: {
      label: 'Slide left',
      run(oldEl, newEl, url, dur) {
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '1';
        newEl.style.transform = 'translateX(100%)';
        void newEl.offsetWidth;
        newEl.style.transition = `transform ${dur}s ease`;
        oldEl.style.transition = `transform ${dur}s ease`;
        newEl.style.transform = 'translateX(0)';
        oldEl.style.transform = 'translateX(-100%)';
        newEl.classList.add('show');
        scheduleCleanup(oldEl, dur * 1000 + 100);
      },
    },
    slideUp: {
      label: 'Slide up',
      run(oldEl, newEl, url, dur) {
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '1';
        newEl.style.transform = 'translateY(100%)';
        void newEl.offsetWidth;
        newEl.style.transition = `transform ${dur}s ease`;
        oldEl.style.transition = `transform ${dur}s ease`;
        newEl.style.transform = 'translateY(0)';
        oldEl.style.transform = 'translateY(-100%)';
        newEl.classList.add('show');
        scheduleCleanup(oldEl, dur * 1000 + 100);
      },
    },
    zoomIn: {
      label: 'Zoom in',
      run(oldEl, newEl, url, dur) {
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;

        b.style.transition = `opacity ${dur / 3}s ease`;
        b.classList.add('show');
        await sleep(dur * 333);
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        void newEl.offsetWidth;
        newEl.style.transition = `opacity ${dur * 0.4}s ease`;
        newEl.style.opacity = '1';
        newEl.classList.add('show');
        oldEl.style.opacity = '0';
        oldEl.classList.remove('show');
        await sleep(50);
        b.classList.remove('show');
        setTimeout(() => { b.style.background = '#000'; }, dur * 1000 + 300);
      },
    },
    burn: {
      label: 'Burn',
      run(oldEl, newEl, url, dur) {
        // Hot warm-yellow tint as old fades; new fades in cool. The
        // contrast reads as a film burn-out without the harshness.
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        void newEl.offsetWidth;
        newEl.style.transition = `opacity ${dur}s ease`;
        newEl.style.opacity = '1';
        newEl.classList.add('show');
        oldEl.style.transition = `opacity ${dur}s ease, filter ${dur}s ease`;
        oldEl.style.filter = 'sepia(1) hue-rotate(-20deg) saturate(2) brightness(1.3)';
        oldEl.style.opacity = '0';
        oldEl.classList.remove('show');
        setTimeout(() => { oldEl.style.filter = ''; }, dur * 1000 + 150);
      },
    },
    vignetteOut: {
      label: 'Vignette',
      run(oldEl, newEl, url, dur) {
        // Old slide darkens from the edges inward (ring closes) while
        // new fades in underneath at full brightness. The mask is a
        // shrinking radial gradient applied via -webkit-mask-image.
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '1';
        newEl.style.zIndex = '1';
        oldEl.style.zIndex = '2';
        oldEl.style.maskImage = 'radial-gradient(circle, black 100%, transparent 100%)';
        oldEl.style.webkitMaskImage = 'radial-gradient(circle, black 100%, transparent 100%)';
        void oldEl.offsetWidth;
        oldEl.style.transition = `mask-image ${dur}s ease, -webkit-mask-image ${dur}s ease, opacity ${dur}s ease`;
        oldEl.style.maskImage = 'radial-gradient(circle, black 0%, transparent 0%)';
        oldEl.style.webkitMaskImage = 'radial-gradient(circle, black 0%, transparent 0%)';
        oldEl.style.opacity = '0';
        newEl.classList.add('show');
        oldEl.classList.remove('show');
        setTimeout(() => {
          oldEl.style.maskImage = '';
          oldEl.style.webkitMaskImage = '';
          oldEl.style.zIndex = '';
          newEl.style.zIndex = '';
        }, dur * 1000 + 100);
      },
    },

    // ─── Combo / kinetic ──────────────────────────────────────────────
    fallIn: {
      label: 'Fall in',
      run(oldEl, newEl, url, dur) {
        // New drops from above with a slight overshoot; old slides
        // down a little while fading. Bouncy entrance, gentle exit.
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '0';
        newEl.style.transform = 'translateY(-30%) scale(1.05)';
        void newEl.offsetWidth;
        newEl.style.transition = `opacity ${dur * 0.6}s ease, transform ${dur}s cubic-bezier(.34, 1.36, .64, 1)`;
        newEl.style.opacity = '1';
        newEl.style.transform = 'translateY(0) scale(1)';
        newEl.classList.add('show');
        oldEl.style.transition = `opacity ${dur}s ease, transform ${dur}s ease-in`;
        oldEl.style.transform = 'translateY(15%)';
        oldEl.style.opacity = '0';
        oldEl.classList.remove('show');
        scheduleCleanup(oldEl, dur * 1000 + 100);
      },
    },
    floatIn: {
      label: 'Float in',
      run(oldEl, newEl, url, dur) {
        // The opposite of fallIn — drifts upward gently. Hot-air-balloon
        // ascending feel: slow, calm, no bounce.
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.opacity = '0';
        newEl.style.transform = 'translateY(20%) scale(0.95)';
        void newEl.offsetWidth;
        newEl.style.transition = `opacity ${dur}s ease, transform ${dur}s ease-out`;
        newEl.style.opacity = '1';
        newEl.style.transform = 'translateY(0) scale(1)';
        newEl.classList.add('show');
        oldEl.style.transition = `opacity ${dur}s ease, transform ${dur}s ease-in`;
        oldEl.style.transform = 'translateY(-15%) scale(1.05)';
        oldEl.style.opacity = '0';
        oldEl.classList.remove('show');
        scheduleCleanup(oldEl, dur * 1000 + 100);
      },
    },
    breathe: {
      label: 'Breathe',
      run(oldEl, newEl, url, dur) {
        // Slow scale-pulse at the start of new slide (1.0 → 1.03 → 1.0)
        // alongside a crossfade. Reads like a quiet inhale/exhale —
        // closest cousin to the kenburns transition but symmetrical.
        prepSlides(oldEl, newEl);
        newEl.style.backgroundImage = `url("${url}")`;
        newEl.style.transform = 'scale(0.97)';
        void newEl.offsetWidth;
        newEl.style.transition = `opacity ${dur}s ease, transform ${dur * 1.5}s ease-in-out`;
        newEl.style.opacity = '1';
        newEl.style.transform = 'scale(1)';
        newEl.classList.add('show');
        oldEl.style.transition = `opacity ${dur}s ease, transform ${dur}s ease-out`;
        oldEl.style.transform = 'scale(1.03)';
        oldEl.style.opacity = '0';
        oldEl.classList.remove('show');
        scheduleCleanup(oldEl, dur * 1500 + 100);
      },
    },
  };

  // ══════════════════════════════════════════════════════════════════════
  // Fullscreen slideshow player
  // ══════════════════════════════════════════════════════════════════════
  const player = {
    running: false,
    slides: [$('slideA'), $('slideB')],
    slot: 0,
    urls: [],
    order: [],
    cursor: 0,
    timer: null,
    autoGenTimer: null,

    async start() {
      const manifest = await getManifest();
      if (!manifest.length) { toast('Gallery is empty. Generate some images first.', 'warn'); return; }
      // Require at least one transition enabled
      if (!getEnabledTransitions().length) {
        toast('Enable at least one transition effect.', 'warn'); return;
      }

      toast('Loading images...');
      this.urls = [];
      for (const e of manifest) {
        const url = await cacheGetUrl(e.id);
        if (url) this.urls.push(url);
      }
      if (!this.urls.length) { toast('Failed to load images.', 'err'); return; }
      this.rebuildOrder();
      this.cursor = 0;
      this.running = true;

      // Clean initial slide state
      this.slides.forEach(s => {
        s.classList.remove('show');
        s.style.cssText = '';
      });
      $('blackout').classList.remove('show');
      $('blackout').style.background = '#000';

      // Enter fullscreen on the player element
      $('player').classList.add('active');
      const el = $('player');
      try {
        if (el.requestFullscreen)             await el.requestFullscreen();
        else if (el.webkitRequestFullscreen)  el.webkitRequestFullscreen();
      } catch { /* user dismissed fullscreen; continue inline */ }

      // First slide — instant cut, no transition
      this.slot = 0;
      const first = this.urls[this.order[0]];
      const s = this.slides[0];
      s.style.backgroundImage = `url("${first}")`;
      s.style.opacity = '1';
      s.style.zIndex = '2';
      s.classList.add('show');
      this.slides[1].style.zIndex = '1';

      const intervalSec = Number($('interval').value);
      this.scheduleNext(intervalSec);

      if ($('autoGen').checked) this.startAutoGen();
    },

    rebuildOrder() {
      this.order = this.urls.map((_, i) => i);
      if (currentOrder === 'random') {
        for (let i = this.order.length - 1; i > 0; i--) {
          const j = Math.floor(Math.random() * (i + 1));
          [this.order[i], this.order[j]] = [this.order[j], this.order[i]];
        }
      }
    },

    scheduleNext(sec) {
      clearTimeout(this.timer);
      this.timer = setTimeout(() => {
        if (!this.running) return;
        this.advance();
      }, sec * 1000);
    },

    async advance() {
      if (!this.running) return;
      this.cursor++;
      if (this.cursor >= this.order.length) {
        // Refresh urls at wrap in case auto-gen added images mid-play
        const manifest = await getManifest();
        const fresh = [];
        for (const e of manifest) {
          const u = await cacheGetUrl(e.id);
          if (u) fresh.push(u);
        }
        if (fresh.length) this.urls = fresh;
        this.rebuildOrder();
        this.cursor = 0;
      }
      const nextUrl = this.urls[this.order[this.cursor]];
      if (!nextUrl) { this.scheduleNext(Number($('interval').value)); return; }

      // Pick a random transition from the enabled set (uniform)
      const enabled = getEnabledTransitions();
      const pick = enabled[Math.floor(Math.random() * enabled.length)] || 'cut';
      const tx = TRANSITIONS[pick] || TRANSITIONS.cut;
      const dur = Number($('duration').value);
      const nextSlot = 1 - this.slot;
      const oldEl = this.slides[this.slot];
      const newEl = this.slides[nextSlot];

      try {
        await Promise.resolve(tx.run(oldEl, newEl, nextUrl, dur));
      } catch (e) {
        console.error('transition failed:', pick, e);
      }

      this.slot = nextSlot;
      this.scheduleNext(Number($('interval').value));
    },

    startAutoGen() {
      clearInterval(this.autoGenTimer);
      this.autoGenTimer = setInterval(async () => {
        if (!this.running) return;
        if (activeGens >= MAX_PARALLEL) return;
        const m = await getManifest();
        if (m.length >= MAX_IMAGES) return;
        generateOne(); // fire-and-forget — picked up at next wrap
      }, Math.max(8, Number($('interval').value) * 2) * 1000);
    },

    stop() {
      if (!this.running) return;
      this.running = false;
      clearTimeout(this.timer);
      clearInterval(this.autoGenTimer);
      this.slides.forEach(s => {
        s.classList.remove('show');
        s.style.cssText = '';
      });
      $('blackout').classList.remove('show');
      $('player').classList.remove('active');
      $('player').classList.remove('flipping'); // a flip transition may have been mid-flight
      if (document.fullscreenElement)        document.exitFullscreen?.();
      else if (document.webkitFullscreenElement) document.webkitExitFullscreen?.();
      renderGallery();
    },
  };

  // ══════════════════════════════════════════════════════════════════════
  // Utilities
  // ══════════════════════════════════════════════════════════════════════
  function $(id) { return document.getElementById(id); }
  function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

  let toastTimer = null;
  function toast(msg, kind = 'info') {
    const el = $('toast');
    el.textContent = msg;
    el.className = 'toast show ' + (kind === 'err' ? 'err' : kind === 'warn' ? 'warn' : '');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(() => el.classList.remove('show'), kind === 'err' ? 6000 : 3000);
  }

  // ══════════════════════════════════════════════════════════════════════
  // Event wiring
  // ══════════════════════════════════════════════════════════════════════
  $('guidance').addEventListener('input', e => { $('guidanceVal').textContent = e.target.value; saveSettings(); });
  $('steps').addEventListener('input',    e => { $('stepsVal').textContent    = e.target.value; saveSettings(); });
  $('interval').addEventListener('input', e => { $('intervalVal').textContent = e.target.value; saveSettings(); });
  $('duration').addEventListener('input', e => { $('durationVal').textContent = Number(e.target.value).toFixed(1); saveSettings(); });

  // Prompts (persist on every keystroke, debounced)
  $('promptPos').addEventListener('input', () => saveSettings());
  $('promptNeg').addEventListener('input', () => saveSettings());

  // Orientation + auto-gen + enhance length selector
  $('orientation').addEventListener('change', () => saveSettings(0));
  $('autoGen').addEventListener('change', () => saveSettings(0));
  $('enhanceLength')?.addEventListener('change', () => saveSettings(0));

  // ✨ Prompt enhancer — toggle-cancel handled inside runEnhance()
  $('enhanceBtn')?.addEventListener('click', runEnhance);

  // Style chips — clear-all wipes selections (individual chips wire
  // themselves inside buildStyleChips() since they're rebuilt on toggle)
  $('chipsClear')?.addEventListener('click', () => {
    if (!activeStyleIds.length) return;
    activeStyleIds = [];
    buildStyleChips();
    saveSettings(0);
  });

  // Playback order — segmented control (Sequential / Random)
  $('orderSeg').addEventListener('click', e => {
    const btn = e.target.closest('button[data-order]');
    if (!btn) return;
    currentOrder = btn.dataset.order;
    $('orderSeg').querySelectorAll('button').forEach(b => b.classList.toggle('active', b === btn));
    saveSettings(0);
  });

  // Transition grid bulk controls
  $('txAll').addEventListener('click', () => {
    document.querySelectorAll('#transitionGrid input[type="checkbox"]').forEach(b => b.checked = true);
    saveSettings(0);
  });
  $('txNone').addEventListener('click', () => {
    document.querySelectorAll('#transitionGrid input[type="checkbox"]').forEach(b => b.checked = false);
    saveSettings(0);
  });
  $('txInvert').addEventListener('click', () => {
    document.querySelectorAll('#transitionGrid input[type="checkbox"]').forEach(b => b.checked = !b.checked);
    saveSettings(0);
  });

  // Individual transition checkbox toggles (delegated so it works even
  // though buildTransitionGrid re-creates the DOM)
  $('transitionGrid').addEventListener('change', e => {
    if (e.target.matches('input[type="checkbox"]')) saveSettings(0);
  });

  // Gallery sort + favorites filter — both persisted, both re-render
  $('gallerySort')?.addEventListener('change', (e) => {
    gallerySort = e.target.value || 'newest';
    saveSettings(0);
    renderGallery();
  });
  $('favFilter')?.addEventListener('click', () => {
    favOnly = !favOnly;
    saveSettings(0);
    renderGallery();
  });

  // Generation buttons — queue-based, hard-capped at MAX_IMAGES
  $('gen1').addEventListener('click', () => _qAdd(1));
  $('gen5').addEventListener('click', () => _qAdd(5));
  $('qRemove')?.addEventListener('click', _qRemoveOne);
  $('qClear') ?.addEventListener('click', _qClearAll);

  // Old delegated .del handler is no longer needed — renderGallery()
  // wires per-item .del / .dl / .fav handlers directly inside the
  // grid. The body click of each item opens the lightbox.

  $('clearAll').addEventListener('click', async () => {
    const m = await getManifest();
    if (!m.length) return;
    if (!confirm(`Delete all ${m.length} images? Favorited ones will be removed too.`)) return;
    await cacheClear();
    renderGallery();
    toast('Gallery cleared.');
  });

  $('downloadAll').addEventListener('click', async () => {
    const m = await getManifest();
    if (!m.length) { toast('Nothing to download.', 'warn'); return; }
    toast(`Downloading ${m.length} image${m.length === 1 ? '' : 's'}…`);
    for (const e of m) {
      await cacheDownload(e.id);
      await sleep(250); // space out so the browser doesn't block
    }
  });

  $('playBtn').addEventListener('click', () => player.start());

  // Exit on click OR ESC (ESC exits fullscreen; we hook fullscreenchange too).
  // The .ignore-click guard prevents the player from closing immediately
  // when the user releases their click on Play Fullscreen — the click
  // bubbles into the fullscreen player otherwise.
  $('player').addEventListener('click', () => player.stop());
  document.addEventListener('keydown', e => {
    if (!player.running) return;
    if (e.key === 'Escape') player.stop();
  });
  document.addEventListener('fullscreenchange', () => {
    if (player.running && !document.fullscreenElement) player.stop();
  });
  document.addEventListener('webkitfullscreenchange', () => {
    if (player.running && !document.webkitFullscreenElement) player.stop();
  });

  // ══════════════════════════════════════════════════════════════════════
  // Initial paint — order matters:
  //   1. Build transition grid (creates checkbox DOM)
  //   2. Build style chips (creates chip DOM — needed before loadSettings
  //      can sync the active state into them)
  //   3. Load persisted settings (populates inputs, checks boxes)
  //   4. Re-render style chips with the loaded active set
  //   5. Render gallery (async, independent of settings)
  //   6. Refresh queue UI (so cap-headroom is reflected immediately)
  //   7. Best-effort: ask browser to keep our IndexedDB around
  // ══════════════════════════════════════════════════════════════════════

  // iOS Safari: stop the browser from auto-zooming when the user focuses
  // a textarea on small touch screens — otherwise the big prompt box
  // yanks the viewport around on every tap.
  try {
    const isTouchScreen = window.matchMedia('(pointer: coarse)').matches;
    const isSafari = navigator.vendor && navigator.vendor.indexOf('Apple') > -1
      && !navigator.userAgent.includes('CriOS')
      && !navigator.userAgent.includes('FxiOS');
    if (isSafari && window.innerWidth < 800 && isTouchScreen) {
      const vp = document.querySelector('[name=viewport]');
      if (vp && !vp.getAttribute('content').includes('maximum-scale')) {
        vp.setAttribute('content', vp.getAttribute('content') + ', maximum-scale=1');
      }
    }
  } catch (e) { console.error(e); }

  (async () => {
    buildTransitionGrid();
    buildStyleChips();
    try { await loadSettings(); } catch (e) { console.error('loadSettings:', e); }
    buildStyleChips(); // re-render after settings populated activeStyleIds
    await renderGallery();
    _bumpQueueUI();
    tryPersistBrowserStorageData();
  })();
})();
