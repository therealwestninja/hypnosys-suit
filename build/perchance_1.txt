// freshSeeds — the API HNE actually calls. Wraps the bank with filter +
// exclude logic. Returns up to `n` phrase strings.
//
// opts.profileTags  — array of tag names derived from the user's profile
//                     (e.g. ['sleep','anxiety']). Phrases match if ANY of
//                     their pipe-separated tags appears in profileTags.
//                     If empty/missing, no tag filter applied (returns
//                     uniformly random phrases).
// opts.n            — how many phrases to return. Default 3. Capped to bank size.
// opts.excludeSet   — array OR Set of phrases to skip (recently-used anchors).
//
// Pure JS — no Perchance list-syntax magic. Returns [] on any error so the
// HTML panel's `if (!phrases || !phrases.length) return ''` guard fires.
freshSeeds.pick = (opts) =>
  try {
    opts = opts || {};
    let bank = root.freshSeedsBank || [];
    if (!bank.length) return [];
    let n = Math.max(1, Math.min(parseInt(opts.n) || 3, bank.length));
    let tags = Array.isArray(opts.profileTags) ? opts.profileTags.map(t => String(t).toLowerCase()) : [];
    let excludeSet = opts.excludeSet instanceof Set
      ? opts.excludeSet
      : new Set(Array.isArray(opts.excludeSet) ? opts.excludeSet : []);

    // Filter bank by tag overlap (ANY-match) + not in excludeSet.
    let candidates = bank.filter(entry => {
      if (!entry || !entry.phrase) return false;
      if (excludeSet.has(entry.phrase)) return false;
      if (!tags.length) return true; // no tag filter — accept all
      let entryTags = String(entry.tag || '').toLowerCase().split('|').map(s => s.trim()).filter(Boolean);
      if (!entryTags.length) return false;
      return entryTags.some(et => tags.includes(et));
    });

    // If the strict tag filter is too restrictive (< n results), back off
    // to ANY phrase not in excludeSet. Better to return some variety than
    // nothing — the AI will adapt regardless of perfect tag match.
    if (candidates.length < n) {
      candidates = bank.filter(entry => entry && entry.phrase && !excludeSet.has(entry.phrase));
    }
    if (!candidates.length) return [];

    // Fisher-Yates shuffle, take first n.
    let pool = candidates.slice();
    for (let i = pool.length - 1; i > 0; i--) {
      let j = Math.floor(Math.random() * (i + 1));
      [pool[i], pool[j]] = [pool[j], pool[i]];
    }
    return pool.slice(0, n).map(e => e.phrase);
  } catch (e) {
    console.warn('freshSeeds.pick failed:', e);
    return [];
  }
