  // ====================================================================
  // IDB STORAGE (ported from Adaptive Session Studio — MIT)
  // Replaces localStorage for reliable persistence

  const IDB_NAME = 'stillness-db';
  const IDB_VER  = 1;
  const IDB_STORE = 'kv';
  let _idb = null;
  let _idbOpenP = null;

  function _openIdb() {
    if (_idb) return Promise.resolve(_idb);
    if (_idbOpenP) return _idbOpenP;
    _idbOpenP = new Promise((resolve, reject) => {
      const req = indexedDB.open(IDB_NAME, IDB_VER);
      req.onupgradeneeded = e => {
        const db = e.target.result;
        if (!db.objectStoreNames.contains(IDB_STORE)) db.createObjectStore(IDB_STORE);
      };

  // IDB-backed storage with localStorage migration

  async function _initStorage() {
    // Migrate personas from localStorage → IDB
    _personaCache = await idbGet(PERSONA_KEY);
    if (!_personaCache) {
      try {
        const lsRaw = localStorage.getItem(PERSONA_KEY);
        if (lsRaw) { _personaCache = JSON.parse(lsRaw); await idbSet(PERSONA_KEY, _personaCache); localStorage.removeItem(PERSONA_KEY); }
      } catch {}
    }
    if (!Array.isArray(_personaCache)) _personaCache = [];

    // Migrate library from localStorage → IDB
    _libraryCache = await idbGet(LIBRARY_KEY);
    if (!_libraryCache) {
      try {
        const lsRaw = localStorage.getItem(LIBRARY_KEY);
        if (lsRaw) { _libraryCache = JSON.parse(lsRaw); await idbSet(LIBRARY_KEY, _libraryCache); localStorage.removeItem(LIBRARY_KEY); }
      } catch {}
    }
    if (!Array.isArray(_libraryCache)) _libraryCache = [];

    // Persist browser storage so IDB isn't cleared
    if (navigator.storage?.persist) await navigator.storage.persist();
  }

  function validateStep(n) {
    if (n === 1) return !!getPersona();
    if (n === 2) return !!getMethod();
    if (n === 3) {
      const skipCraft = ['trance', 'sleep'].includes(state.intentionType);
      return !!state.craftedSuggestion || skipCraft;
    }
    if (n === 4) return true;
    if (n === 5) return !!state.fullScript;
    return true;
  }
 
  function goToStep(n) {
    if (n < 1 || n > TOTAL_STEPS) return;
    // Forward direction: validate current step
    if (n > state.currentStep) {
      for (let i = state.currentStep; i < n; i++) {
        if (!validateStep(i)) { return; }
        state.stepsCompleted[i] = true;
      }
    }
    state.currentStep = n;
    // Stop all previews when leaving the Configure step
    if (typeof _stopAllPreviews === 'function') _stopAllPreviews();
    updateStepUI();
    if (n === 4) { renderConfigSummary(); }
    if (n === 5) { refreshSummary(); }
    window.scrollTo({ top: 0, behavior: 'smooth' });

    // Accessibility: announce step change and move focus
    const stepNames = { 1: 'Choose Your Guide', 2: 'Choose Your Method', 3: 'Set Your Intention', 4: 'Configure the Session', 5: 'Generate Your Session' };
    const announce = document.getElementById('ariaStepAnnounce');
    if (announce) announce.textContent = `Step ${n} of ${TOTAL_STEPS}: ${stepNames[n] || ''}`;
    // Move focus to step heading
    const heading = document.querySelector(`.step-content[data-step="${n}"] .step-title`);
    if (heading) { heading.setAttribute('tabindex', '-1'); heading.focus(); }
  }
 
  function updateStepUI() {
    // Content
    document.querySelectorAll('.step-content').forEach(el => {
      el.classList.toggle('active', parseInt(el.dataset.step) === state.currentStep);
    });
    // Stepper circles
    document.querySelectorAll('.step-item').forEach(el => {
      const n = parseInt(el.dataset.step);
      el.classList.toggle('current', n === state.currentStep);
      el.classList.toggle('completed', n < state.currentStep);
      el.classList.toggle('clickable', n < state.currentStep);
    });
    // Connectors
    document.querySelectorAll('.step-connector').forEach(el => {
      const [from] = el.dataset.connector.split('-').map(Number);
      el.classList.toggle('completed', from < state.currentStep);
    });
    // Step values under circles
    document.getElementById('stepValue1').textContent = state.stepsCompleted[1] ? getPersona().name : '';
    document.getElementById('stepValue2').textContent = state.stepsCompleted[2] ? getMethod().name : '';
    let s3 = '';
    if (state.stepsCompleted[3]) {
      if (state.craftedSuggestion) s3 = 'crafted ✓';
      else if (state.intentionType === 'trance') s3 = 'pure trance';
    }
    document.getElementById('stepValue3').textContent = s3;
    document.getElementById('stepValue4').textContent = state.stepsCompleted[4] ? state.length : '';
    document.getElementById('stepValue5').textContent = state.fullScript ? 'ready ✓' : '';
    // Nav
    document.getElementById('btnBack').hidden = state.currentStep === 1;
    document.getElementById('btnNext').hidden = state.currentStep === TOTAL_STEPS;
    document.getElementById('btnNext').disabled = !validateStep(state.currentStep);
  }

  class SoundscapeEngine {
    constructor() { this.ctx = null; this.master = null; this.compressor = null; this.nodes = []; this.running = false; this.type = 'drone'; this._duckGain = null; }

    async start(volume = 0.35, type = 'drone') {
      if (this.running) await this.stop();
      this.type = type;
      this.ctx = new (window.AudioContext || window.webkitAudioContext)();
      if (this.ctx.state === 'suspended') await this.ctx.resume();

      // Master chain: sceneGain → compressor → destination
      this.compressor = this.ctx.createDynamicsCompressor();
      this.compressor.threshold.value = -18; this.compressor.knee.value = 24;
      this.compressor.ratio.value = 3; this.compressor.attack.value = 0.01; this.compressor.release.value = 0.25;
      this.compressor.connect(this.ctx.destination);
      this._duckGain = this.ctx.createGain(); this._duckGain.gain.value = 1.0;
      this._duckGain.connect(this.compressor);
      this.master = this.ctx.createGain(); this.master.gain.value = 0;
      this.master.connect(this._duckGain);

      const c = this.ctx, m = this.master, ns = this.nodes;
      const mkNoise = (filterType, freq, q) => {
        const buf = c.createBuffer(1, 2 * c.sampleRate, c.sampleRate);
        const d = buf.getChannelData(0); let lo = 0;
        for (let i = 0; i < d.length; i++) {
          const w = Math.random() * 2 - 1;
          if (filterType === 'brown') { d[i] = (lo + 0.02 * w) / 1.02; lo = d[i]; d[i] *= 3.5; }
          else d[i] = w;
        }
        const src = c.createBufferSource(); src.buffer = buf; src.loop = true;
        if (filterType !== 'white' && filterType !== 'brown') {
          const f = c.createBiquadFilter(); f.type = filterType; f.frequency.value = freq || 1000; if (q) f.Q.value = q;
          src.connect(f); ns.push(src, f); return f;
        }
        ns.push(src); return src;
      };
      const mkOsc = (freq, type = 'sine') => { const o = c.createOscillator(); o.type = type; o.frequency.value = freq; ns.push(o); return o; };
      const mkGain = (v) => { const g = c.createGain(); g.gain.value = v; ns.push(g); return g; };
      const connect = (...chain) => { for (let i = 0; i < chain.length - 1; i++) chain[i].connect(chain[i + 1]); };
      const startAll = () => ns.forEach(n => { try { n.start && n.start(); } catch(e) {} });

      if (type === 'drone') {
        const bass = mkOsc(70); const bg = mkGain(0.4); connect(bass, bg, m);
        const harm = mkOsc(141); const hg = mkGain(0.18); connect(harm, hg, m);
        const noise = mkNoise('brown'); const ng = mkGain(0.25); const nf = c.createBiquadFilter(); nf.type = 'lowpass'; nf.frequency.value = 280; ns.push(nf);
        connect(noise, nf, ng, m);
        const lfo = mkOsc(0.07); const lg = mkGain(0.08); connect(lfo, lg); lg.connect(bg.gain);
      } else if (type === 'rain') {
        const bed = mkNoise('bandpass', 1200, 0.8); const bg = mkGain(0.5); connect(bed, bg, m);
        const rumble = mkNoise('brown'); const rf = c.createBiquadFilter(); rf.type = 'lowpass'; rf.frequency.value = 200; ns.push(rf);
        const rg = mkGain(0.08); connect(rumble, rf, rg, m);
        // Droplet scheduler
        const droplet = () => { if (!this.running) return;
          const osc = c.createOscillator(); osc.frequency.value = 2000 + Math.random() * 4000;
          const g = c.createGain(); g.gain.value = 0; osc.connect(g).connect(m);
          const t = c.currentTime; g.gain.setValueAtTime(0, t); g.gain.linearRampToValueAtTime(0.15 + Math.random() * 0.1, t + 0.005);
          g.gain.exponentialRampToValueAtTime(0.0001, t + 0.02 + Math.random() * 0.015);
          osc.start(t); osc.stop(t + 0.04);
          setTimeout(droplet, 20 + Math.random() * 60);
        };

      } else if (type === 'ocean') {

        const lfo = mkOsc(1 / 11); const lg = mkGain(0.35); connect(lfo, lg); lg.connect(ng.gain);
        connect(noise, ng, m);

      } else if (type === 'fire') {

        const bf = c.createBiquadFilter(); bf.type = 'lowpass'; bf.frequency.value = 400; ns.push(bf);
        connect(bed, bf, bg, m);
        const crackle = () => { if (!this.running) return;
          const b = c.createBuffer(1, Math.floor(c.sampleRate * 0.008), c.sampleRate); const d = b.getChannelData(0);
          for (let i = 0; i < d.length; i++) d[i] = (Math.random() * 2 - 1) * (1 - i / d.length);
          const s = c.createBufferSource(); s.buffer = b;
          const f = c.createBiquadFilter(); f.type = 'bandpass'; f.frequency.value = 1500 + Math.random() * 3000; f.Q.value = 2;
          const g = c.createGain(); g.gain.value = 0.15 + Math.random() * 0.2;
          s.connect(f).connect(g).connect(m); s.start();
          setTimeout(crackle, 80 + Math.random() * 300);
        };

      } else if (type === 'bowls') {
        const ratios = [1, 2.02, 3.07, 4.16, 5.38]; const decays = [10, 7, 5, 4, 3];
        const strike = () => { if (!this.running) return;

          ratios.forEach((r, i) => {
            const o = c.createOscillator(); o.frequency.value = root * r;
            const g = c.createGain(); g.gain.value = 0.12 / (i + 1);
            o.connect(g).connect(m); const t = c.currentTime;
            g.gain.setValueAtTime(g.gain.value, t); g.gain.exponentialRampToValueAtTime(0.0001, t + decays[i]);
            o.start(t); o.stop(t + decays[i] + 0.1);
          });
          setTimeout(strike, 8000 + Math.random() * 12000);
        };
        strike();

          [0, 0.14].forEach(delay => {
            const o = c.createOscillator(); o.frequency.value = 55;
            const g = c.createGain(); g.gain.value = 0;
            const f = c.createBiquadFilter(); f.type = 'lowpass'; f.frequency.value = 200;
            o.connect(f).connect(g).connect(m); const t = c.currentTime + delay;
            g.gain.setValueAtTime(0, t); g.gain.linearRampToValueAtTime(0.5, t + 0.03);
            g.gain.exponentialRampToValueAtTime(0.0001, t + 0.25);
            o.start(t); o.stop(t + 0.3);
          });
          setTimeout(beat, 1050 + Math.random() * 120);
        };
        beat();
      } else if (type === 'binaural') {
        const carrier = 200, beatHz = 6; // Theta range
        const oL = mkOsc(carrier - beatHz / 2); const oR = mkOsc(carrier + beatHz / 2);
        const gL = mkGain(0.3); const gR = mkGain(0.3);
        const merger = c.createChannelMerger(2); ns.push(merger);
        oL.connect(gL).connect(merger, 0, 0); oR.connect(gR).connect(merger, 0, 1);
        merger.connect(m);
      } else if (type === 'pink' || type === 'brown' || type === 'white') {
        const noise = mkNoise(type === 'pink' ? 'lowpass' : type, type === 'pink' ? 800 : undefined);
        const ng = mkGain(type === 'white' ? 0.15 : 0.35); connect(noise, ng, m);
      }

      startAll();
      this.master.gain.linearRampToValueAtTime(volume, this.ctx.currentTime + 4);
      this.running = true;
    }

    setVolume(v) { if (this.master && this.ctx) this.master.gain.linearRampToValueAtTime(Math.max(0, v), this.ctx.currentTime + 0.5); }

    // Ducking: called from speakScript to lower ambience during narration
    duck() { if (this._duckGain && this.ctx) this._duckGain.gain.linearRampToValueAtTime(0.2, this.ctx.currentTime + 0.4); }
    unduck() { if (this._duckGain && this.ctx) this._duckGain.gain.linearRampToValueAtTime(1.0, this.ctx.currentTime + 0.8); }

    setPhaseFreq(phase) {
      if (this.type !== 'drone' || !this.ctx) return;
      const freqMap = { settling: 70, induction: 67, deepener: 63, work: 60, emergence: 70 };
      const target = freqMap[phase] || 70;
      if (this.nodes[0]?.frequency) this.nodes[0].frequency.linearRampToValueAtTime(target, this.ctx.currentTime + 4);
      if (this.nodes[2]?.frequency) this.nodes[2].frequency.linearRampToValueAtTime(target * 2.01, this.ctx.currentTime + 4);
    }

    async stop() {
      if (!this.running || !this.ctx) return;
      this.master.gain.linearRampToValueAtTime(0, this.ctx.currentTime + 2);
      await new Promise(r => setTimeout(r, 2100));
      this.nodes.forEach(n => { try { n.stop?.(); n.disconnect?.(); } catch(e) {} });
      try { this.ctx.suspend(); } catch(e) {}
      this.ctx = null; this.master = null; this._duckGain = null; this.compressor = null; this.nodes = []; this.running = false;
    }
  }

  function hexToRgb(hex) {
    let h = hex.replace('#','');

    const r = parseInt(h.slice(0,2),16), g = parseInt(h.slice(2,4),16), b = parseInt(h.slice(4,6),16);
    return `${Number.isFinite(r)?r:128},${Number.isFinite(g)?g:128},${Number.isFinite(b)?b:128}`;
  }

  // Moiré — historically the most potent trance device (per ROADMAP)
  // Two concentric-ring patterns with one center slowly orbiting the other
  function renderMoire(ctx, W, H, t, color) {
    const cx = W/2, cy = H/2, maxR = Math.max(W, H) * 0.8;
    ctx.fillStyle = '#000'; ctx.fillRect(0, 0, W, H);
    ctx.strokeStyle = 'rgba(136,136,136,0.45)'; ctx.lineWidth = 1;
    // First grating: centered
    for (let r = 6; r < maxR; r += 12) {
      ctx.beginPath(); ctx.arc(cx, cy, r, 0, Math.PI * 2); ctx.stroke();
    }
    // Second grating: orbiting offset
    const ox = cx + Math.cos(t * 0.15) * 18;
    const oy = cy + Math.sin(t * 0.12) * 14;
    ctx.strokeStyle = 'rgba(136,136,136,0.4)';
    for (let r = 6; r < maxR; r += 12) {
      ctx.beginPath(); ctx.arc(ox, oy, r, 0, Math.PI * 2); ctx.stroke();
    }
    // Soft vignette
    const vig = ctx.createRadialGradient(cx, cy, maxR * 0.2, cx, cy, maxR * 0.6);
    vig.addColorStop(0, 'transparent'); vig.addColorStop(1, 'rgba(0,0,0,0.7)');
    ctx.fillStyle = vig; ctx.fillRect(0, 0, W, H);
  }

  // Starfield warp tunnel — 3D stars projected to 2D with streak lines
  function renderStarfield(ctx, W, H, t, color) {
    const cx = W/2, cy = H/2, STARS = 200, focal = 180, rgb = hexToRgb(color);
    if (!renderStarfield._pool) {
      renderStarfield._pool = [];
      for (let i = 0; i < STARS; i++) renderStarfield._pool.push({
        x: (Math.random() - 0.5) * 600, y: (Math.random() - 0.5) * 600,
        z: Math.random() * 600, psx: 0, psy: 0
      });
    }
    ctx.fillStyle = 'rgba(0,0,0,0.2)'; ctx.fillRect(0, 0, W, H);
    const speed = 1.5 + Math.sin(t * 0.3) * 0.5;
    for (const s of renderStarfield._pool) {
      const psx = s.psx, psy = s.psy;
      s.z -= speed;
      if (s.z < 1) { s.x = (Math.random()-0.5)*600; s.y = (Math.random()-0.5)*600; s.z = 600; s.psx = cx; s.psy = cy; continue; }
      const sx = s.x / s.z * focal + cx;
      const sy = s.y / s.z * focal + cy;
      s.psx = sx; s.psy = sy;
      if (sx < 0 || sx > W || sy < 0 || sy > H) continue;
      const brightness = Math.min(1, (600 - s.z) / 400);
      const size = Math.max(0.5, (1 - s.z / 600) * 3);
      ctx.fillStyle = `rgba(${rgb},${brightness})`;
      // Streak line from previous position
      if (psx > 0 && Math.abs(sx - psx) < 40) {
        ctx.beginPath(); ctx.moveTo(psx, psy); ctx.lineTo(sx, sy);
        ctx.strokeStyle = `rgba(${rgb},${brightness * 0.5})`; ctx.lineWidth = size * 0.5; ctx.stroke();
      }
      ctx.fillRect(sx - size/2, sy - size/2, size, size);
    }
  }

  // Candle focal — soft radial glow with position jitter (classic Mesmerist's candle)
  function renderCandle(ctx, W, H, t, color) {
    const cx = W/2, cy = H * 0.4, rgb = hexToRgb(color);
    ctx.fillStyle = 'rgba(0,0,0,0.08)'; ctx.fillRect(0, 0, W, H);
    // Flame jitter
    const jx = cx + (Math.random() - 0.5) * 4;
    const jy = cy + (Math.random() - 0.5) * 3;
    const baseR = 30 + Math.sin(t * 2.5) * 5 + Math.sin(t * 4.1) * 3;
    // Outer glow
    const outer = ctx.createRadialGradient(jx, jy, 0, jx, jy, baseR * 3);
    outer.addColorStop(0, `rgba(${rgb},0.15)`);
    outer.addColorStop(0.4, `rgba(${rgb},0.05)`);
    outer.addColorStop(1, 'transparent');
    ctx.fillStyle = outer; ctx.fillRect(0, 0, W, H);
    // Inner flame
    const inner = ctx.createRadialGradient(jx, jy - 4, 0, jx, jy, baseR);
    inner.addColorStop(0, 'rgba(255,240,200,0.7)');
    inner.addColorStop(0.3, `rgba(${rgb},0.4)`);
    inner.addColorStop(0.7, `rgba(${rgb},0.08)`);
    inner.addColorStop(1, 'transparent');
    ctx.beginPath(); ctx.arc(jx, jy, baseR, 0, Math.PI * 2);
    ctx.fillStyle = inner; ctx.fill();
    // Bright core
    ctx.beginPath(); ctx.arc(jx, jy - 2, 4, 0, Math.PI * 2);
    ctx.fillStyle = 'rgba(255,255,240,0.9)'; ctx.fill();
  }

  // Flow-field particles — simplex noise driven, additive glow
  // Uses a simplified 2D noise (value noise with smoothstep) since we can't import simplex
  function renderFlowField(ctx, W, H, t, color) {
    const rgb = hexToRgb(color);
    const N = W < 500 ? 250 : 600; // Mobile vs desktop
    if (!renderFlowField._p) {
      renderFlowField._p = [];
      for (let i = 0; i < N; i++) renderFlowField._p.push({ x: Math.random()*W, y: Math.random()*H, px: 0, py: 0 });
    }
    const ps = renderFlowField._p;
    // Resize pool if canvas changed
    if (ps.length < N) while (ps.length < N) ps.push({ x: Math.random()*W, y: Math.random()*H, px: 0, py: 0 });

    ctx.fillStyle = 'rgba(0,0,0,0.04)'; ctx.fillRect(0, 0, W, H);
    ctx.globalCompositeOperation = 'lighter';

    const scale = 0.003, speed = 0.5;
    for (const p of ps) {
      p.px = p.x; p.py = p.y;
      // Simple hash-based noise field
      const nx = p.x * scale, ny = p.y * scale, nt = t * 0.08;
      const angle = (Math.sin(nx * 3.1 + nt) + Math.cos(ny * 2.7 - nt * 0.7) + Math.sin((nx + ny) * 1.4 + nt * 0.3)) * Math.PI;
      p.x += Math.cos(angle) * speed;
      p.y += Math.sin(angle) * speed;
      // Wrap around
      if (p.x < 0) p.x += W; if (p.x > W) p.x -= W;
      if (p.y < 0) p.y += H; if (p.y > H) p.y -= H;
      // Draw streak
      const dx = p.x - p.px, dy = p.y - p.py;
      if (Math.abs(dx) < 20 && Math.abs(dy) < 20) {
        ctx.beginPath(); ctx.moveTo(p.px, p.py); ctx.lineTo(p.x, p.y);
        ctx.strokeStyle = `rgba(${rgb},0.3)`; ctx.lineWidth = 0.8; ctx.stroke();
      }
    }
    ctx.globalCompositeOperation = 'source-over';
    // Vignette
    const vig = ctx.createRadialGradient(W/2, H/2, Math.min(W,H)*0.2, W/2, H/2, Math.min(W,H)*0.6);
    vig.addColorStop(0, 'transparent'); vig.addColorStop(1, 'rgba(0,0,0,0.4)');
    ctx.fillStyle = vig; ctx.fillRect(0, 0, W, H);
  }

  // Classic plasma — sum of sines mapped through HSV palette
  function renderPlasma(ctx, W, H, t, color) {
    // Render at reduced resolution for performance
    const rW = 192, rH = Math.round(rW * H / W);
    if (!renderPlasma._img || renderPlasma._img.width !== rW) {
      renderPlasma._off = new OffscreenCanvas ? new OffscreenCanvas(rW, rH) : document.createElement('canvas');
      renderPlasma._off.width = rW; renderPlasma._off.height = rH;
      renderPlasma._img = renderPlasma._off.getContext('2d').createImageData(rW, rH);
      // Precompute sine LUT
      renderPlasma._sin = new Float32Array(1024);
      for (let i = 0; i < 1024; i++) renderPlasma._sin[i] = Math.sin(i * Math.PI * 2 / 1024);
    }
    const data = renderPlasma._img.data, sinLut = renderPlasma._sin;
    const S = v => sinLut[((v % 1024) + 1024) % 1024 | 0];

    for (let y = 0; y < rH; y++) {
      for (let x = 0; x < rW; x++) {
        const v = S(x * 3 + t * 40) + S(y * 4 - t * 30) + S((x + y) * 2 + t * 20) + S(Math.sqrt(x*x + y*y) * 2 - t * 50);
        const hue = ((v + 4) / 8 * 360 + t * 15) % 360;
        // HSV to RGB (S=0.7, V=0.6)
        const c = 0.42, x2 = c * (1 - Math.abs((hue / 60) % 2 - 1));
        let r1=0,g1=0,b1=0;
        const h = Math.floor(hue / 60) % 6;
        if (h===0){r1=c;g1=x2;}else if(h===1){r1=x2;g1=c;}else if(h===2){g1=c;b1=x2;}
        else if(h===3){g1=x2;b1=c;}else if(h===4){r1=x2;b1=c;}else{r1=c;b1=x2;}
        const m = 0.18;
        const idx = (y * rW + x) * 4;
        data[idx] = (r1+m)*255; data[idx+1] = (g1+m)*255; data[idx+2] = (b1+m)*255; data[idx+3] = 255;
      }
    }
    const offCtx = renderPlasma._off.getContext('2d');
    offCtx.putImageData(renderPlasma._img, 0, 0);
    ctx.imageSmoothingEnabled = true;
    ctx.drawImage(renderPlasma._off, 0, 0, W, H);
  }

  // Kaleidoscope — wraps any sub-renderer with N-fold mirror symmetry
  function makeKaleidoscope(subRenderer, folds = 6) {
    return function(ctx, W, H, t, color) {
      const cx = W/2, cy = H/2;
      // Render sub-effect to an offscreen canvas
      if (!makeKaleidoscope._off || makeKaleidoscope._off.width !== W) {
        makeKaleidoscope._off = document.createElement('canvas');
        makeKaleidoscope._off.width = W; makeKaleidoscope._off.height = H;
      }
      const offCtx = makeKaleidoscope._off.getContext('2d');
      subRenderer(offCtx, W, H, t, color);
      // Draw N wedges with alternating mirrors
      ctx.clearRect(0, 0, W, H);
      const wedgeAngle = Math.PI * 2 / folds;
      for (let i = 0; i < folds; i++) {
        ctx.save();
        ctx.translate(cx, cy);
        ctx.rotate(wedgeAngle * i);
        if (i % 2 === 1) ctx.scale(-1, 1); // Mirror alternate wedges
        ctx.beginPath();
        ctx.moveTo(0, 0);
        ctx.lineTo(Math.max(W, H), 0);
        ctx.arc(0, 0, Math.max(W, H), 0, wedgeAngle);
        ctx.closePath();
        ctx.clip();
        ctx.translate(-cx, -cy);
        ctx.drawImage(makeKaleidoscope._off, 0, 0);
        ctx.restore();
      }
    };
  }

  function parseScript(text) {
    const phases = { settling: '', induction: '', deepener: '', work: '', emergence: '' };
    const markers = [
      { key: 'settling', rx: /===\s*SETTLING\s*===/i },
      { key: 'induction', rx: /===\s*INDUCTION\s*===/i },
      { key: 'deepener', rx: /===\s*DEEPENER\s*===/i },
      { key: 'work', rx: /===\s*WORK\s*===/i },
      { key: 'emergence', rx: /===\s*EMERGENCE\s*===/i },
    ];
    const positions = markers.map(m => { const match = text.match(m.rx); return { key: m.key, index: match ? match.index + match[0].length : -1, start: match ? match.index : -1 }; });
    const found = positions.filter(p => p.index !== -1).sort((a,b) => a.index - b.index);
    for (let i = 0; i < found.length; i++) {
      const cur = found[i];
      const end = i + 1 < found.length ? found[i+1].start : text.length;
      phases[cur.key] = text.substring(cur.index, end).trim();
    }
    return phases;
  }

  async function renderSessionHistory() {
    const panel = document.getElementById('sessionStatsPanel');
    const list = document.getElementById('sessionHistoryList');
    if (!panel || !list) return;

    const history = (await idbGet('stillness_history_v1')) || [];
    if (history.length === 0) {

      return;
    }

    // Calculate stats
    const totalSessions = history.length;
    const totalMinutes = Math.round(history.reduce((sum, s) => sum + (s.durationSec || 0), 0) / 60);
    const completed = history.filter(s => s.completed).length;

    // Streak calculation
    const today = new Date(); today.setHours(0,0,0,0);
    const dayMs = 86400000;
    const sessionDays = [...new Set(history.map(s => {
      const d = new Date(s.date); d.setHours(0,0,0,0); return d.getTime();
    }))].sort((a,b) => b - a);

    let streak = 0;
    let checkDay = today.getTime();
    for (const day of sessionDays) {
      if (day === checkDay || day === checkDay - dayMs) {
        streak++;
        checkDay = day;
      } else break;
    }

    // Favorite guide
    const guideCounts = {};
    history.forEach(s => { guideCounts[s.guide] = (guideCounts[s.guide] || 0) + 1; });
    const favGuide = Object.entries(guideCounts).sort((a,b) => b[1] - a[1])[0]?.[0] || '—';

    panel.innerHTML = `
      <div class="stats-bar">
        <div class="stat-chip"><span class="stat-num">${totalSessions}</span><span class="stat-label">sessions</span></div>
        <div class="stat-chip"><span class="stat-num">${totalMinutes}</span><span class="stat-label">minutes</span></div>
        <div class="stat-chip"><span class="stat-num">${streak}</span><span class="stat-label">day streak</span></div>
        <div class="stat-chip"><span class="stat-num">${Math.round(completed/totalSessions*100)}%</span><span class="stat-label">completed</span></div>
      </div>
      <div style="font-size:0.72rem;color:var(--text-faint);text-align:center;">favourite guide: ${escapeHtml(favGuide)}</div>`;

    // Recent sessions list (last 10)
    list.innerHTML = '';
    history.slice(0, 10).forEach(s => {
      const d = new Date(s.date);
      const dateStr = d.toLocaleDateString(undefined, { month: 'short', day: 'numeric' });
      const mins = Math.round((s.durationSec || 0) / 60);
      const item = document.createElement('div');
      item.className = 'history-item';
      item.innerHTML = `<span class="h-date">${dateStr}</span><span class="h-guide">${escapeHtml(s.guide || '—')} · ${escapeHtml(s.method || '—')}</span><span class="h-dur">${mins}m${s.completed ? '' : ' ✕'}</span>`;
      list.appendChild(item);
    });
  }

  async function shareConfig() {

  async function loadSharedConfig() {

    if (!hash.startsWith('#cfg=')) return;
    const encoded = hash.slice(5);
    let json;
    try {
      const bytes = Uint8Array.from(atob(encoded), c => c.charCodeAt(0));
      const stream = new Blob([bytes]).stream().pipeThrough(new DecompressionStream('gzip'));
      json = await new Response(stream).text();

    try {
      const cfg = JSON.parse(json);
      if (cfg.p && state.personas.find(p => p.id === cfg.p)) state.selectedPersonaId = cfg.p;

        if (!c.classList.contains('fixation-dot') && !c.classList.contains('audio-indicator')
          && !c.classList.contains('persona-indicator') && !c.classList.contains('method-indicator')

      });
    } else {
      setupVisual(vizChoice);
    }
    populateTranscript();
    document.getElementById('transcriptPanel')?.classList.remove('visible');
    const tBtn = document.getElementById('transcriptToggle');
    if (tBtn) tBtn.textContent = '📜 script';

    // Start soundscape (replaces old drone toggle)
    if (state.soundscape && state.soundscape !== 'none') {
      try {
        await drone.start(state.droneVolume, state.soundscape);
        document.getElementById('audioIndicator').textContent = '· ' + state.soundscape + ' ·';
      } catch(e) {
        notify.info('Audio unavailable — tap the screen to enable audio, then try again.');
      }
    }

    // Haptic marker — double-pulse on session start (Android/Chrome)
    vibrateSessionMarker();

    // Wake Lock — prevent screen timeout during session
    try {
      if (navigator.wakeLock) {
        state._wakeLock = await navigator.wakeLock.request('screen');
        state._wakeLock.addEventListener('release', () => { state._wakeLock = null; });
      }
    } catch(e) { /* Wake Lock not available or denied */ }

    // Content warning for advanced/intense methods
    const needsWarning = method.category === 'advanced' || ['drill-instructor','dominatrix','interrogator','commander'].includes(getPersona().id);
    if (needsWarning) {
      const warningShown = await new Promise(resolve => {

    state._timeInterval = setInterval(() => {
      if (state.paused) return;

      const mins = Math.floor(elapsed / 60);
      const secs = elapsed % 60;

      if (timeEl) timeEl.textContent = `${mins}:${String(secs).padStart(2,'0')} elapsed`;
    }, 1000);

    const phases = ['settling','induction','deepener','work','emergence'];
    const phaseLabels = { settling: 'settling', induction: 'going inward', deepener: 'deeper', work: method.category === 'demonstration' ? 'the effect' : 'transformation', emergence: 're-alerting' };

    for (let pi = 0; pi < phases.length; pi++) {
      const phase = phases[pi];
      if (state.abortFlag) break;
      const phaseText = state.parsedScript[phase];
      if (!phaseText || phaseText.trim().length < 10) continue;

      // Update phase state
      state._currentPhase = phase;
      document.getElementById('phaseIndicator').textContent = phaseLabels[phase];

      // Update phase dots
      document.querySelectorAll('.phase-dot').forEach((d, i) => {
        if (i < pi) d.classList.add('done');

      // Update progress bar

      document.getElementById('sessionProgressFill').style.width = pct + '%';

      // Modulate drone frequency per phase
      if (state.droneEnabled && drone.running) drone.setPhaseFreq(phase);

      const cleanText = phaseText.replace(/^===\s*\w+\s*===\s*/i, '').trim();
      await speakScript(cleanText);
    }

    // Complete progress bar
    document.getElementById('sessionProgressFill').style.width = '100%';
    document.querySelectorAll('.phase-dot').forEach(d => { d.classList.remove('active'); d.classList.add('done'); });

    if (!state.abortFlag) finishSession(false);
  }

  // Interruptible sleep — checks abortFlag & paused state in 200ms chunks so
  // pause/end during a [PAUSE n] marker responds quickly instead of waiting
  // for the whole sleep to finish.
  async function interruptibleSleep(totalMs, bypassAbort = false) {
    const CHUNK = 200;
    let elapsed = 0;
    while (elapsed < totalMs) {
      if (!bypassAbort && state.abortFlag) return;
      while (state.paused) { await sleep(CHUNK); }
      const remaining = totalMs - elapsed;
      await sleep(Math.min(CHUNK, remaining));
      elapsed += CHUNK;
    }
  }

  async function speakScript(text, bypassAbort = false) {
    // Sanitize: strip any malformed [pause...] or [PAUSE...] markers that
    // wouldn't match our canonical regex. Prevents TTS from reading "bracket
    // pause bracket" aloud.
    const canonical = text.replace(/\[\s*pause[^\]]*\]/gi, (m) => {
      const num = m.match(/\d+/);
      return num ? `[PAUSE ${num[0]}]` : '';
    });

    const parts = canonical.split(/\[PAUSE\s+(\d+)\]/i);
    const display = document.getElementById('textDisplay');
    for (let i = 0; i < parts.length; i++) {
      if (!bypassAbort && state.abortFlag) return;
      while (state.paused) await sleep(200);
      if (i % 2 === 0) {
        const segment = parts[i].trim();
        if (!segment) continue;
        const chunks = segment.split(/(?<=[.!?])\s+/).filter(s => s.trim());
        for (const chunk of chunks) {
          if (!bypassAbort && state.abortFlag) return;
          while (state.paused) await sleep(200);
          display.style.opacity = 0;
          await sleep(350);
          display.textContent = chunk;
          display.style.opacity = 0.5;
          if (drone.running) drone.duck();
          await speak(chunk);
          if (drone.running) drone.unduck();
        }
      } else {
        const seconds = parseInt(parts[i], 10) || 4;
        await interruptibleSleep(seconds * 1000, bypassAbort);
      }
    }
  }

  function speak(text) {
    return new Promise(resolve => {
      if (!('speechSynthesis' in window)) { setTimeout(resolve, text.length * 80); return; }
      if (speechSynthesis.speaking && !speechSynthesis.paused) {
        speechSynthesis.cancel();
        setTimeout(() => _doSpeak(text, resolve), 50);
      } else {
        _doSpeak(text, resolve);
      }
    });
  }
  function _doSpeak(text, resolve) {
    const u = new SpeechSynthesisUtterance(text);
    u.voice = state.voice || getSelectedVoice();
    u.rate = state.speechRate; u.pitch = state.voicePitch || 0.85; u.volume = state.voiceVolume ?? 1.0;
    let settled = false;
    const done = () => { if (!settled) { settled = true; resolve(); } };
    u.onend = done;
    u.onerror = done;

    // Word-level highlighting via onboundary (Chrome/Safari desktop)
    const display = document.getElementById('textDisplay');
    u.onboundary = (e) => {
      if (e.name === 'word' && display) {
        const before = text.slice(0, e.charIndex);
        const word = text.slice(e.charIndex, e.charIndex + e.charLength);
        const after = text.slice(e.charIndex + e.charLength);
        display.innerHTML = `<span style="opacity:0.35">${escapeHtml(before)}</span><span style="opacity:1;color:var(--text)">${escapeHtml(word)}</span><span style="opacity:0.35">${escapeHtml(after)}</span>`;
      }
    };

    speechSynthesis.speak(u);

    // Double induction: fire a second utterance with a different voice simultaneously
    if (state._doubleInduction && state._secondVoice) {
      const u2 = new SpeechSynthesisUtterance(text);
      u2.voice = state._secondVoice;
      u2.rate = state.speechRate * 0.92; // Slightly offset tempo
      u2.pitch = 0.75; u2.volume = 0.7;
      u2.onend = () => {}; u2.onerror = () => {};
      speechSynthesis.speak(u2);
    }

    const maxWait = Math.max(8000, text.length * 200);
    setTimeout(() => { if (!settled && !speechSynthesis.speaking) done(); }, maxWait);
    setTimeout(done, Math.max(15000, text.length * 250));
  }
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  let _recorder = null, _recordedChunks = [];

  function startRecording() {
    if (!drone.ctx || !drone.master) { notify.warn('Start a session with a soundscape first.'); return; }
    try {
      const dest = drone.ctx.createMediaStreamDestination();
      drone.master.connect(dest);
      _recorder = new MediaRecorder(dest.stream, { mimeType: 'audio/webm;codecs=opus' });
      _recordedChunks = [];
      _recorder.ondataavailable = e => { if (e.data.size > 0) _recordedChunks.push(e.data); };
      _recorder.onstop = () => {
        if (_recordedChunks.length === 0) return;
        const blob = new Blob(_recordedChunks, { type: 'audio/webm' });
        const a = document.createElement('a');
        a.href = URL.createObjectURL(blob);
        a.download = `stillness-ambience-${Date.now()}.webm`;
        a.click(); URL.revokeObjectURL(a.href);
        notify.success('Ambience recording downloaded.');
        _recordedChunks = [];
      };
      _recorder.start();
      notify.info('Recording ambience...');
      const btn = document.getElementById('recordBtn');
      if (btn) { btn.textContent = '⏹ stop recording'; btn.classList.add('recording'); }
    } catch(e) { notify.error('Recording not supported in this browser.'); }
  }

  function stopRecording() {
    if (_recorder && _recorder.state === 'recording') {
      _recorder.stop(); _recorder = null;
      const btn = document.getElementById('recordBtn');
      if (btn) { btn.textContent = '⏺ record ambience'; btn.classList.remove('recording'); }
    }
  }

  function toggleRecording() {
    if (_recorder && _recorder.state === 'recording') stopRecording();
    else startRecording();
  }

  async function renderHeatmap() {
    const wrap = document.getElementById('heatmapWrap');
    if (!wrap) return;
    const history = (await idbGet('stillness_history_v1')) || [];
    if (history.length === 0) { wrap.innerHTML = ''; return; }

    // Build day→count map for the last 16 weeks (112 days)
    const dayCounts = {};
    history.forEach(s => {
      const d = new Date(s.date); d.setHours(0,0,0,0);
      const key = d.toISOString().slice(0,10);
      dayCounts[key] = (dayCounts[key] || 0) + 1;
    });

    const today = new Date(); today.setHours(0,0,0,0);
    const dayMs = 86400000;
    // Start from 15 weeks ago, aligned to Monday
    const startDay = new Date(today.getTime() - 15 * 7 * dayMs);
    startDay.setDate(startDay.getDate() - ((startDay.getDay() + 6) % 7)); // Monday

    let html = '<div class="heatmap">';
    let d = new Date(startDay);
    while (d <= today) {
      html += '<div class="heatmap-col">';
      for (let row = 0; row < 7 && d <= today; row++) {
        const key = d.toISOString().slice(0,10);
        const count = dayCounts[key] || 0;
        const level = count === 0 ? 0 : count === 1 ? 1 : count === 2 ? 2 : count <= 4 ? 3 : 4;
        const title = `${key}: ${count} session${count !== 1 ? 's' : ''}`;
        html += `<div class="heatmap-cell" data-count="${level}" title="${title}"></div>`;
        d.setDate(d.getDate() + 1);
      }
      html += '</div>';
    }
    html += '</div>';
    wrap.innerHTML = html;
  }
